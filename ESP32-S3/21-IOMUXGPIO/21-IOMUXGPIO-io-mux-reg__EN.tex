\begin{register}{H}{IO\_MUX\_PIN\_CTRL\_REG}{0x{}0000}\label{regdesc:IOMUXPINCTRL}
\regfield{(reserved)}{16}{16}{{0x{}0}}%
\regfield{IO\_MUX\_PAD\_POWER\_CTRL}{1}{15}{{0x{}0}}%
\regfield{IO\_MUX\_SWITCH\_PRT\_NUM}{3}{12}{{0x{}2}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK3}{4}{8}{{0x{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK2}{4}{4}{{0x{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK1}{4}{0}{{0x{}0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXPINCTRLCLKX}\item [IO\_MUX\_PIN\_CTRL\_CLK\regindex{x}] If you want to output clock for I2S0 to: (R/W)\\
CLK\_OUT1, then set IO\_MUX\_PIN\_CTRL\_CLK1 = 0x{}0;\\
CLK\_OUT2, then set IO\_MUX\_PIN\_CTRL\_CLK2 = 0x{}0;\\
CLK\_OUT3, then set IO\_MUX\_PIN\_CTRL\_CLK3 = 0x{}0. \\

If you want to output clock for I2S1 to: (R/W)\\
CLK\_OUT1, then set IO\_MUX\_PIN\_CTRL\_CLK1 = 0x{}F;\\
CLK\_OUT2, then set IO\_MUX\_PIN\_CTRL\_CLK2 = 0x{}F;\\
CLK\_OUT3, then set IO\_MUX\_PIN\_CTRL\_CLK3 = 0x{}F. \\

{\bfseries Note:}\\
Only the above mentioned combinations of clock source and clock output pins are possible. \\
The CLK\_OUT1 \verb+~+ 3 can be found in \hyperref[tab:iomuxgpio-pin-functions]{IO\_MUX Pin Function List}.
\item [IO\_MUX\_SWITCH\_PRT\_NUM] GPIO pin power switch delay, delay unit is one APB clock. (R/W)
\label{fielddesc:IOMUXPADPOWERCTRL}\item [IO\_MUX\_PAD\_POWER\_CTRL] Select power voltage for GPIO33 \verb+~+ 37 and GPIO47 \verb+~+ 48. 1: select VDD\_SPI 1.8 V; 0: select VDD3P3\_CPU 3.3 V. (R/W)

\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{IO\_MUX\_GPIO\regindex{n}\_REG (\regindex{n}: 0-21, 26-48)}{0x{}0004+4*\regindex{n}}\label{regdesc:IOMUXnREG}
\regfield{(reserved)}{16}{16}{0000000000000000}%
\regfield{IO\_MUX\_FILTER\_EN}{1}{15}{{0x{}0}}%
\regfield{IO\_MUX\_MCU\_SEL}{3}{12}{{0x{}0}}%
\regfield{IO\_MUX\_FUN\_DRV}{2}{10}{{0x{}2}}%
\regfield{IO\_MUX\_FUN\_IE}{1}{9}{0}%
\regfield{IO\_MUX\_FUN\_WPU}{1}{8}{0}%
\regfield{IO\_MUX\_FUN\_WPD}{1}{7}{0}%
\regfield{IO\_MUX\_MCU\_DRV}{2}{5}{{00}}%
\regfield{IO\_MUX\_MCU\_IE}{1}{4}{0}%
\regfield{IO\_MUX\_MCU\_WPU}{1}{3}{0}%
\regfield{IO\_MUX\_MCU\_WPD}{1}{2}{0}%
\regfield{IO\_MUX\_SLP\_SEL}{1}{1}{0}%
\regfield{IO\_MUX\_MCU\_OE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXMCUOE}\item [IO\_MUX\_MCU\_OE] Output enable of the pin in sleep mode. 1: Output enabled; 0: Output disabled. (R/W)
\label{fielddesc:IOMUXSLPSEL}\item [IO\_MUX\_SLP\_SEL] Sleep mode selection of this pin. Set to 1 to put the pin in sleep mode. (R/W)
\label{fielddesc:IOMUXMCUWPD}\item [IO\_MUX\_MCU\_WPD] Pull-down enable of the pin during sleep mode. 1: Internal pull-down enabled; 0: Internal pull-down disabled. (R/W)
\label{fielddesc:IOMUXMCUWPU}\item [IO\_MUX\_MCU\_WPU] Pull-up enable of the pin during sleep mode. 1: Internal pull-up enabled; 0: Internal pull-up disabled.
\label{fielddesc:IOMUXMCUIE}\item [IO\_MUX\_MCU\_IE] Input enable of the pin during sleep mode. 1: Input enabled; 0: Input disabled. (R/W)
\label{fielddesc:IOMUXMCUDRV}\item [IO\_MUX\_MCU\_DRV] Configures the drive strength of GPIO\regindex{n} during sleep mode.
\begin{itemize}
    \item GPIO17 and GPIO18\\
        0: \verb+~+5 mA\\
        1: \verb+~+\textbf{20 mA}\\
        2: \verb+~+\textbf{10 mA}\\
        3: \verb+~+40 mA\\
    \item Other GPIOs\\
        0: \verb+~+5 mA\\
        1: \verb+~+\textbf{10 mA}\\
        2: \verb+~+\textbf{20 mA}\\
        3: \verb+~+40 mA\\
    \end{itemize}
    (R/W)
\label{fielddesc:IOMUXFUNWPD}\item [IO\_MUX\_FUN\_WPD] Pull-down enable of the pin. 1: Pull-down enabled; 0: Pull-down disabled. (R/W)
\label{fielddesc:IOMUXFUNWPU}\item [IO\_MUX\_FUN\_WPU] Pull-up enable of the pin. 1: Internal pull-up enabled; 0: Internal pull-up disabled. (R/W)
\label{fielddesc:IOMUXFUNIE}\item [IO\_MUX\_FUN\_IE] Input enable of the pin. 1: Input enabled; 0: Input disabled. (R/W)
\item [Continued on the next page...]
\end{reglist}\end{regdesc}
\end{register}
