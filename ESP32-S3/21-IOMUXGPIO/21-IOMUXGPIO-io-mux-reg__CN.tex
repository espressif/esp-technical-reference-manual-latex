\begin{register}{H}{IO\_MUX\_PIN\_CTRL\_REG}{0x{}0000}\label{regdesc:IOMUXPINCTRL}
\regfield{(reserved)}{16}{16}{{0x{}0}}%
\regfield{IO\_MUX\_PAD\_POWER\_CTRL}{1}{15}{{0x{}0}}%
\regfield{IO\_MUX\_SWITCH\_PRT\_NUM}{3}{12}{{0x{}2}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK3}{4}{8}{{0x{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK2}{4}{4}{{0x{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK1}{4}{0}{{0x{}0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXPINCTRLCLKX}\item [IO\_MUX\_PIN\_CTRL\_CLK\regindex{x}] 配置 I2S0 外设时钟输出到：\\
CLK\_OUT1，配置 IO\_MUX\_PIN\_CTRL\_CLK1 = 0x{}0；\\
CLK\_OUT2，配置 IO\_MUX\_PIN\_CTRL\_CLK2 = 0x{}0；\\
CLK\_OUT3，配置 IO\_MUX\_PIN\_CTRL\_CLK3 = 0x{}0。\\(R/W)

配置 I2S1 外设时钟输出到：\\
CLK\_OUT1，配置 IO\_MUX\_PIN\_CTRL\_CLK1 = 0x{}F；\\
CLK\_OUT2，配置 IO\_MUX\_PIN\_CTRL\_CLK2 = 0x{}F；\\
CLK\_OUT3，配置 IO\_MUX\_PIN\_CTRL\_CLK3 = 0x{}F。\\(R/W)
{\bfseries 说明：}\\
只能有上述配置组合。\\
CLK\_OUT1 \verb+~+ 3 可在 \hyperref[tab:iomuxgpio-pin-functions]{IO MUX 管脚功能列表}中查询。
\label{fielddesc:IOMUXSWITCHPRTNUM}\item [IO\_MUX\_SWITCH\_PRT\_NUM] GPIO 管脚电源切换延时，延时单位为一个 APB 时钟周期。(R/W)
\label{fielddesc:IOMUXPADPOWERCTRL}\item [IO\_MUX\_PAD\_POWER\_CTRL] 选择 GPIO33 \verb+~+ 37 和 GPIO47 \verb+~+ 48 的电源电压。1：选择 VDD\_SPI 1.8 V 供电；0：选择 VDD3P3\_CPU 3.3 V 供电。(R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{IO\_MUX\_GPIO\regindex{n}\_REG (\regindex{n}: 0-21, 26-48)}{0x{}0004+4*\regindex{n}}\label{regdesc:IOMUXnREG}
\regfield{(reserved)}{16}{16}{0000000000000000}%
\regfield{IO\_MUX\_FILTER\_EN}{1}{15}{{0x{}0}}%
\regfield{IO\_MUX\_MCU\_SEL}{3}{12}{{0x{}0}}%
\regfield{IO\_MUX\_FUN\_DRV}{2}{10}{{0x{}2}}%
\regfield{IO\_MUX\_FUN\_IE}{1}{9}{0}%
\regfield{IO\_MUX\_FUN\_WPU}{1}{8}{0}%
\regfield{IO\_MUX\_FUN\_WPD}{1}{7}{0}%
\regfield{IO\_MUX\_MCU\_DRV}{2}{5}{{00}}%
\regfield{IO\_MUX\_MCU\_IE}{1}{4}{0}%
\regfield{IO\_MUX\_MCU\_WPU}{1}{3}{0}%
\regfield{IO\_MUX\_MCU\_WPD}{1}{2}{0}%
\regfield{IO\_MUX\_SLP\_SEL}{1}{1}{0}%
\regfield{IO\_MUX\_MCU\_OE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXMCUOE}\item [IO\_MUX\_MCU\_OE] 睡眠模式下管脚的输出使能。1：输出使能；0：输出关闭。(R/W)
\label{fielddesc:IOMUXSLPSEL}\item [IO\_MUX\_SLP\_SEL] 管脚的睡眠模式选择。置 1 将使能睡眠模式。(R/W)
\label{fielddesc:IOMUXMCUWPD}\item [IO\_MUX\_MCU\_WPD] 睡眠模式下管脚的下拉使能。1：内部下拉使能；0：内部下拉关闭。(R/W)
\label{fielddesc:IOMUXMCUWPU}\item [IO\_MUX\_MCU\_WPU] 睡眠模式下管脚的上拉使能。1：内部上拉使能；0：内部上拉关闭。(R/W)
\label{fielddesc:IOMUXMCUIE}\item [IO\_MUX\_MCU\_IE] 睡眠模式下管脚的输入使能。1：输入使能；0：输入关闭。(R/W)
\label{fielddesc:IOMUXMCUDRV}\item [IO\_MUX\_MCU\_DRV] 配置睡眠模式下 GPIO\regindex{n} 的驱动强度。
\begin{itemize}
\item GPIO17 和 GPIO18\\
    0: \verb+~+5 mA\\
    1: \verb+~+\textbf{20 mA}\\
    2: \verb+~+\textbf{10 mA}\\
    3: \verb+~+40 mA\\
\item 其他 GPIO\\
    0: \verb+~+5 mA\\
    1: \verb+~+\textbf{10 mA}\\
    2: \verb+~+\textbf{20 mA}\\
    3: \verb+~+40 mA\\
\end{itemize}
(R/W)
\label{fielddesc:IOMUXFUNWPD}\item [IO\_MUX\_FUN\_WPD] 管脚的下拉使能。1：内部下拉使能；0：内部下拉关闭。(R/W)
\label{fielddesc:IOMUXFUNWPU}\item [IO\_MUX\_FUN\_WPU] 管脚的上拉使能。1：内部上拉使能；0：内部上拉关闭。(R/W)
\label{fielddesc:IOMUXFUNIE}\item [IO\_MUX\_FUN\_IE] 管脚的输入使能。1：输入使能；0：输入关闭。(R/W)
\item [见下页]
\end{reglist}\end{regdesc}
\end{register}
