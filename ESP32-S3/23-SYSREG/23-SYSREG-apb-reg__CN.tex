\begin{register}{H}{SYSCON\_CLKGATE\_FORCE\_ON\_REG}{0x{}00A8}\label{regdescSYSCONCLKGATEFORCEONREG}
\regfield{(reserved)}{18}{14}{000000000000000000}%
\regfield{SYSCON\_SRAM\_CLKGATE\_FORCE\_ON}{11}{3}{{0x{}7ff}}%
\regfield{SYSCON\_ROM\_CLKGATE\_FORCE\_ON}{3}{0}{{0x{}7}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddescSYSCONROMCLKGATEFORCEON}\item [SYSCON\_ROM\_CLKGATE\_FORCE\_ON] 置 1 配置 ROM 内存的时钟门控始终打开；置 0 则配置 ROM 内存的时钟门控在被访问时自动打开，没有访问时自动关闭。(R/W)
\label{fielddescSYSCONSRAMCLKGATEFORCEON}\item [SYSCON\_SRAM\_CLKGATE\_FORCE\_ON] 置 1 配置 SRAM 内存的时钟门控始终打开；置 0 则配置 SRAM 内存的时钟门控在被访问时自动打开，没有访问时自动关闭。(R/W)
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{SYSCON\_MEM\_POWER\_DOWN\_REG}{0x{}00AC}\label{regdescSYSCONMEMPOWERDOWNREG}
\regfield{(reserved)}{18}{14}{000000000000000000}%
\regfield{SYSCON\_SRAM\_POWER\_DOWN}{11}{3}{{0}}%
\regfield{SYSCON\_ROM\_POWER\_DOWN}{3}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddescSYSCONROMPOWERDOWN}\item [SYSCON\_ROM\_POWER\_DOWN] 控制 Internal ROM 进入 Retention 状态。(R/W)
\label{fielddescSYSCONSRAMPOWERDOWN}\item [SYSCON\_SRAM\_POWER\_DOWN] 控制 Internal SRAM 进入 Retention 状态。(R/W)
\end{reglist}\end{regdesc}
\end{register}

\regfield{SYSCON\_ROM\_POWER\_UP}{3}{0}{{0x{}7}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}

\end{reglist}\end{regdesc}
\end{register}
