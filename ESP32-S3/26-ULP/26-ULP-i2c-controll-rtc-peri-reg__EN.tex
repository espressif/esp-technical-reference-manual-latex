\regfield{SENS\_SAR\_I2C\_START}{1}{28}{{0}}%
\regfield{SENS\_SAR\_I2C\_WR\_EN}{1}{27}{{0}}%
\regfield{SENS\_SAR\_I2C\_WDATA}{8}{19}{{0}}%
\regfield{SENS\_SAR\_I2C\_REG\_ADDR}{8}{11}{{0}}%

\begin{regdesc}\begin{reglist}
\label{fielddesc:SENSSARI2CSLAVEADDR}\item [SENS\_SAR\_I2C\_SLAVE\_ADDR] Configures the slave address. (R/W)
\label{fielddesc:SENSSARI2CSLAVEADDR10BITEN}\item [SENS\_SAR\_I2C\_SLAVE\_ADDR\_10BIT\_EN] Configures whether to expand the slave address to 10 bits.\\
0: Not expand\\
1: Expand\\
(R/W)
\label{fielddesc:SENSSARI2CREGADDR}\item [SENS\_SAR\_I2C\_REG\_ADDR] Configures the register address. (R/W)
\label{fielddesc:SENSSARI2CWDATA}\item [SENS\_SAR\_I2C\_WDATA] Configures the data to write. (R/W)
\label{fielddesc:SENSSARI2CWREN}\item [SENS\_SAR\_I2C\_WR\_EN] Configures whether to use write command.\\
0: Use read command\\
1: Use write command\\
(R/W)
\label{fielddesc:SENSSARI2CSTART}\item [SENS\_SAR\_I2C\_START] Start RTC I2C; active only when \hyperref[fielddesc:SENSSARI2CSTARTFORCE]{SENS\_SAR\_I2C\_START\_FORCE} = 1. (R/W)
\label{fielddesc:SENSSARI2CSTARTFORCE}\item [SENS\_SAR\_I2C\_START\_FORCE] 0: RTC I2C started by FSM; 1: RTC I2C started by software. (R/W)
\end{reglist}\end{regdesc}
\end{register}
