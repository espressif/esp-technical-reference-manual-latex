\begin{register}{H}{XTS\_AES\_LINESIZE\_REG}{0x{}0040}\label{regdesc:XTSAESLINESIZEREG}
\regfield{(reserved)}{30}{2}{{0x{}00000000}}%
\regfield{XTS\_AES\_LINESIZE}{2}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_LINESIZE] 块大小寄存器，决定单次加密的数据量。
    \begin{itemize}
        \item 0：加密 16 bytes；
        \item 1：加密 32 bytes；
        \item 2：加密 64 bytes。(R/W)
    \end{itemize}
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{XTS\_AES\_DESTINATION\_REG}{0x{}0044}\label{regdesc:XTSAESDESTINATIONREG}
\regfield{(reserved)}{31}{1}{{0x{}00000000}}%
\regfield{XTS\_AES\_DESTINATION}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_DESTINATION] 决定手动加密类型，目前只能手动加密 flash，所以只能为 0。用户不能写入 1，否则将发生错误。0：加密 flash；1：加密片外 RAM。(R/W)

\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{XTS\_AES\_STATE\_REG}{0x{}0058}\label{regdesc:XTSAESSTATEREG}
\regfield{(reserved)}{30}{2}{{0x{}00000000}}%
\regfield{XTS\_AES\_STATE}{2}{0}{{0x{}0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_STATE] 手动加密模块状态寄存器。
    \begin{itemize}
        \item 0x{}0 (XTS\_AES\_IDLE)：空闲；
        \item 0x{}1 (XTS\_AES\_BUSY)：忙于计算；
        \item 0x{}2 (XTS\_AES\_DONE)：计算完成，但手动加密结果数据对 SPI 不可见；
        \item 0x{}3 (XTS\_AES\_RELEASE)：手动加密结果对 SPI 可见。(RO)
    \end{itemize}
\end{reglist}\end{regdesc}
\end{register}
