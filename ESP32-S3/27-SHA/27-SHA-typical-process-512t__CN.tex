\begin{enumerate}
    \item \label{others:sha-sel-algorithm2} 选择运算标准。
    \begin{itemize}
        \item 配置 \hyperref[regdesc:SHAMODEREG]{SHA\_MODE\_REG} 寄存器为 7 选择 SHA-512/{\it{t}} 运算标准。
    \end{itemize}
    \item 计算哈希初始值。
        \begin{enumerate}
        \item 配置 t\_string 和 t\_length，并将其写入 \hyperref[regdesc:SHATSTRINGREG]{SHA\_T\_STRING\_REG} 和 \hyperref[regdesc:SHATLENGTHREG]{SHA\_T\_LENGTH\_REG} 寄存器。具体请见 \ref{sec:sha-set-initial-hash} 章节。
        \item 对 \hyperref[regdesc:SHASTARTREG]{SHA\_START\_REG} 寄存器置 1，启动 SHA 加速器的运算。
        \item 轮询寄存器 \hyperref[regdesc:SHABUSYREG]{SHA\_BUSY\_REG} 一直到读回的值为 0，代表 SHA 硬件加速器已完成哈希初始值的计算。
    \end{enumerate}
    
    \item \label{others:sha-current-block1} 处理当前信息块\hyperref[others:sha-note-typical-sha]{$^1$}。
    \begin{itemize}
        \item 将当前信息块写入 \hyperref[regdesc:SHAMNREG]{SHA\_M\_\regindex{n}\_REG} 寄存器。
    \end{itemize}
    \item 启动 SHA 加速器。
    \begin{itemize}
        \item 对 \hyperref[regdesc:SHACONTINUEREG]{SHA\_CONTINUE\_REG} 寄存器置 1，启动 SHA 加速器的运算。此时，SHA 加速器使用 \hyperref[regdesc:SHAHNREG]{SHA\_H\_\regindex{n}\_REG} 寄存器中的值作为哈希初始值进行运算。
    \end{itemize}
    
    \item 查询当前信息块的处理进度。
    \begin{itemize}
        \item 轮询寄存器 \hyperref[regdesc:SHABUSYREG]{SHA\_BUSY\_REG} 一直到读回的值为 0，代表 SHA 硬件加速器已完成对当前信息块的计算，进入“空闲”状态\hyperref[others:sha-note-typical-sha]{$^3$}。 
    \end{itemize}
    
    \item 选择是否有后续的待处理信息块。
    \begin{itemize}
        \item 如果存在后续待处理信息块，则跳回执行步骤 \ref{others:sha-current-block1}。
        \item 否则，继续执行。
    \end{itemize}
    \item 获取信息摘要：
    \begin{itemize}
        \item 从寄存器堆 \hyperref[regdesc:SHAHNREG]{SHA\_H\_\regindex{n}\_REG} 取出信息摘要。
    \end{itemize}
\end{enumerate}
