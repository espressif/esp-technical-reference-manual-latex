 \textbf{模块} & \textbf{配置参数/选项} \\ \hline

 \rowcolor{lightgray}
 \textbf{模块} & \textbf{配置参数/选项}\\
 \hline

\multirow{10}{*}{PWM 生成器} &
    \vspace{-1em}
    \begin{itemize}
        \item 设置 PWM\textcolor{red}{\it{x}}A 和/或 PWM\textcolor{red}{\it{x}}B 输出的 PWM 占空比
        \item 设置定时事件发生的时间
        \item 设置发生定时事件时采取的行动：
        \begin{itemize}
           \item 改变 PWM\textcolor{red}{\it{x}}A 和/或 PWM\textcolor{red}{\it{x}}B 输出为高或低
           \item 将 PWM\textcolor{red}{\it{x}}A 和/或 PWM\textcolor{red}{\it{x}}B 取反
           \item 不对输出执行任何操作
        \end{itemize}
        \item 通过直接软件控制强制 PWM 输出的状态
        \item 在 PWM 输出的上升和/或下降边沿上增加死区
        \item 配置该模块的更新方式
     \vspace{-1em}\end{itemize} \\\hline
     
\multirow{7}{*}{死区生成器}  &
    \vspace{-1em}\begin{itemize}
        \item 控制高侧和低侧开关之间的互补死区关系
        \item 指定上升沿死区
        \item 指定下降沿死区
        \item 绕过死区发生器模块，PWM 波形不插入死区
        \item 可根据 PWM\textcolor{red}{\it{x}}A 输出进行 PWM\textcolor{red}{\it{x}}B 相移
        \item 配置该模块的更新方式
    \vspace{-1em}\end{itemize}\\\hline
    
    \multirow{5}{*}{PWM 载波} &
    \vspace{-1em}\begin{itemize}
        \item 使能载波，设置载波频率
        \item 设置载波波形中第一个脉冲的持续时间
        \item 设置第二个以及之后的脉冲的占空比
        \item 绕过 PWM 载波模块，PWM 波形无变动
    \vspace{-1em}\end{itemize}\\\hline
    
    \multirow{13}{*}{故障处理器} &
    \vspace{-1em}\begin{itemize}    
        \item 配置 PWM 模块是否以及如何响应故障事件信号
        \item 指定发生故障事件时采取的操作：
        \begin{itemize}
            \item 强制 PWM\textcolor{red}{\it{x}}A 和/或 PWM\textcolor{red}{\it{x}}B 为高电平
            \item 强制 PWM\textcolor{red}{\it{x}}A 和/或 PWM\textcolor{red}{\it{x}}B 为低电平
            \item 配置 PWM\textcolor{red}{\it{x}}A 和/或 PWM\textcolor{red}{\it{x}}B 忽略任何故障事件
        \end{itemize}
        \item 配置 PWM 应对故障事件的间隔模式：
        \begin{itemize}
            \item 一次性模式
            \item 逐周期模式
        \end{itemize}
        \item 生成中断
        \item 绕过故障处理器模块
        \item 设置逐周期操作清除的方式
        \item 当时基计数器采取向上和向下时，可采取不同操作
\vspace{-1em}
    \end{itemize}\\\hline
