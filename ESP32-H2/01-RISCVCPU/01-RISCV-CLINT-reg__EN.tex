\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:CLINTMTIMECTLMTCE}\item [MTCE] Configures whether to enable the CLINT timer counter.\\
0: Not enable\\
1: Enable\\
(R/W)
\label{fielddesc:CLINTMTIMECTLMTIE}\item [MTIE] Write 1 to enable the machine timer interrupt. (R/W)
\label{fielddesc:CLINTMTIMECTLMTIP}\item [MTIP] Represents the pending status of the machine timer interrupt.\\
0: Not pending\\
1: Pending\\
(RO)
\label{fielddesc:CLINTMTIMECTLMTOF}\item [MTOF] Configures whether the machine timer overflows.\\
0: Not overflow\\
1: Overflow\\
(R/W)
\end{reglist}
\end{regdesc}
\end{register}
