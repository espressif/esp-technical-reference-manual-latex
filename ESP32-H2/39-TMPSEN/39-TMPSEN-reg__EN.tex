\begin{register}{H}{APB\_SARADC\_APB\_TSENS\_CTRL\_REG}{0x0058}\label{regdesc:APBSARADCAPBTSENSCTRLREG}
\regfield{(reserved)}{9}{23}{000000000}%
\regfield{APB\_SARADC\_TSENS\_PU}{1}{22}{{0}}%
\regfield{APB\_SARADC\_TSENS\_CLK\_DIV}{8}{14}{{6}}%
\regfield{APB\_SARADC\_TSENS\_IN\_INV}{1}{13}{{0}}%
\regfield{(reserved)}{5}{8}{00000}%
\regfield{APB\_SARADC\_TSENS\_OUT}{8}{0}{{0x80}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:APBSARADCTSENSOUT}\item [APB\_SARADC\_TSENS\_OUT] Stores the temperature sensor output value. (RO)
\label{fielddesc:APBSARADCTSENSININV}\item [APB\_SARADC\_TSENS\_IN\_INV] Configures whether to invert temperature sensor data. \\
0: No effect\\
1: Invert\\
(R/W)
\label{fielddesc:APBSARADCTSENSCLKDIV}\item [APB\_SARADC\_TSENS\_CLK\_DIV] Configures temperature sensor clock division. (R/W)
\label{fielddesc:APBSARADCTSENSPU}\item [APB\_SARADC\_TSENS\_PU] Configures whether to power up temperature sensor. \\
0: No effect\\
1: Power up\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{APB\_SARADC\_APB\_TSENS\_CTRL2\_REG}{0x005C}\label{regdesc:APBSARADCAPBTSENSCTRL2REG}
\regfield{(reserved)}{16}{16}{0000000000000000}%
\regfield{APB\_SARADC\_TSENS\_CLK\_SEL}{1}{15}{{0}}%
\regfield{APB\_SARADC\_TSENS\_CLK\_INV}{1}{14}{{0x0}}%
\regfield{APB\_SARADC\_TSENS\_XPD\_FORCE}{2}{12}{{0x0}}%
\regfield{APB\_SARADC\_TSENS\_XPD\_WAIT}{12}{0}{{0x2}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:APBSARADCTSENSCLKSEL}\item [APB\_SARADC\_TSENS\_CLK\_SEL] Configures the working clock for temperature sensor. \\
0: RC\_FAST\_CLK\\
1: XTAL\_CLK\\
(R/W)
\label{fielddesc:APBSARADCTSENSCLKINV}\item [APB\_SARADC\_TSENS\_CLK\_INV] Configures the phase of the temperature sensor clock. (R/W)
\label{fielddesc:APBSARADCTSENSXPDFORCE}\item [APB\_SARADC\_TSENS\_XPD\_FORCE] Configures whether to enable force power up/down the temperature sensor. \\
0: Disable force power up function\\
1: Disable force power down function\\
2: Enable force power up temperature sensor\\
3: Enable force power down temperature sensor\\
(R/W)
\label{fielddesc:APBSARADCTSENSXPDWAIT}\item [APB\_SARADC\_TSENS\_XPD\_WAIT] Configure the wait time (in clock cycles) for the senor to release from reset. (R/W)
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{APB\_TSENS\_WAKE\_REG}{0x0064}\label{regdesc:APBTSENSWAKEREG}
\regfield{(reserved)}{13}{19}{0000000000000}%
\regfield{APB\_SARADC\_WAKEUP\_EN}{1}{18}{{0}}%
\regfield{APB\_SARADC\_WAKEUP\_MODE}{1}{17}{{0}}%
\regfield{APB\_SARADC\_WAKEUP\_OVER\_UPPER\_TH}{1}{16}{{0}}%
\regfield{APB\_SARADC\_WAKEUP\_TH\_HIGH}{8}{8}{{0xff}}%
\regfield{APB\_SARADC\_WAKEUP\_TH\_LOW}{8}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:APBSARADCWAKEUPTHLOW}\item [APB\_SARADC\_WAKEUP\_TH\_LOW] Configures the low threshold for temperature monitoring wake-up function. (R/W)
\label{fielddesc:APBSARADCWAKEUPTHHIGH}\item [APB\_SARADC\_WAKEUP\_TH\_HIGH] Configures the high threshold for temperature monitoring wake-up function. (R/W)
\label{fielddesc:APBSARADCWAKEUPOVERUPPERTH}\item [APB\_SARADC\_WAKEUP\_OVER\_UPPER\_TH] Represents whether the temperature output value exceeds the threshold.\\
0: The temperature output value is below the low threshold\\
1: The temperature output value is above the high threshold\\
(RO)
\label{fielddesc:APBSARADCWAKEUPMODE}\item [APB\_SARADC\_WAKEUP\_MODE] Selects the wake-up mode for temperature monitoring. \\
0: Absolute value mode\\
1: Change value mode\\
(R/W)
\label{fielddesc:APBSARADCWAKEUPEN}\item [APB\_SARADC\_WAKEUP\_EN] Configures whether to enable temperature monitoring wake-up function. \\
0: Disable\\
1: Enable\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}
