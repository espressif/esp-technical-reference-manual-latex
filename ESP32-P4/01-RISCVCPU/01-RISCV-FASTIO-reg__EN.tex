\begin{register}{H}{cpu\_gpio\_oen}{0x803}\label{regdesc:GPIO_OEN}
\regfield{(reserved)}{24}{8}{{0}}%
\regfield{CPU\_GPIO\_OEN[7]}{1}{7}{{0}}%
\regfield{CPU\_GPIO\_OEN[6]}{1}{6}{{0}}%
\regfield{CPU\_GPIO\_OEN[5]}{1}{5}{{0}}%
\regfield{CPU\_GPIO\_OEN[4]}{1}{4}{{0}}%
\regfield{CPU\_GPIO\_OEN[3]}{1}{3}{{0}}%
\regfield{CPU\_GPIO\_OEN[2]}{1}{2}{{0}}%
\regfield{CPU\_GPIO\_OEN[1]}{1}{1}{{0}}%
\regfield{CPU\_GPIO\_OEN[0]}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:GPIO_OEN}\item [CPU\_GPIO\_OEN] Configures whether to enable GPIOn (n=0 \verb+~+ 21) output. CPU\_GPIO\_OEN[7:0] correspond to output enable signals cpu\_gpio\_out\_oen[7:0] in Table \ref{tab:iomuxgpio-periph-signals-via-hp-gpio-matrix-en} \textit{Peripheral Signals via GPIO Matrix}. CPU\_GPIO\_OEN value matches that of cpu\_gpio\_out\_oen.
CPU\_GPIO\_OEN is the enable signal of \hyperref[fielddesc:GPIO_OUT]{CPU\_GPIO\_OUT}.\\
0: Disable GPIO output\\
1: Enable GPIO output\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{cpu\_gpio\_in}{0x804}\label{regdesc:GPIO_IN}
\regfield{(reserved)}{24}{8}{{0}}%
\regfield{CPU\_GPIO\_IN[7]}{1}{7}{{0}}%
\regfield{CPU\_GPIO\_IN[6]}{1}{6}{{0}}%
\regfield{CPU\_GPIO\_IN[5]}{1}{5}{{0}}%
\regfield{CPU\_GPIO\_IN[4]}{1}{4}{{0}}%
\regfield{CPU\_GPIO\_IN[3]}{1}{3}{{0}}%
\regfield{CPU\_GPIO\_IN[2]}{1}{2}{{0}}%
\regfield{CPU\_GPIO\_IN[1]}{1}{1}{{0}}%
\regfield{CPU\_GPIO\_IN[0]}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:GPIO_IN}\item [CPU\_GPIO\_IN] Represents GPIOn (n=0 \verb+~+ 21) input value. It is a CPU CSR to read input value (1=high, 0=low) from SoC GPIO pin.\\
CPU\_GPIO\_IN[7:0] correspond to input signals cpu\_gpio\_in[7:0] in Table \ref{tab:iomuxgpio-periph-signals-via-hp-gpio-matrix-en} \textit{Peripheral Signals via GPIO Matrix} .\\
CPU\_GPIO\_IN[7:0] can only be mapped to GPIO pins through GPIO matrix. For details please refer to Section \ref{sec:peripheral-input-via-gpio-matrix} in Chapter \textit{\nameref{mod:iomuxgpio}}.\\
(RO)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{cpu\_gpio\_out}{0x805}\label{regdesc:GPIO_OUT}
\regfield{(reserved)}{24}{8}{{0}}%
\regfield{CPU\_GPIO\_OUT[7]}{1}{7}{{0}}%
\regfield{CPU\_GPIO\_OUT[6]}{1}{6}{{0}}%
\regfield{CPU\_GPIO\_OUT[5]}{1}{5}{{0}}%
\regfield{CPU\_GPIO\_OUT[4]}{1}{4}{{0}}%
\regfield{CPU\_GPIO\_OUT[3]}{1}{3}{{0}}%
\regfield{CPU\_GPIO\_OUT[2]}{1}{2}{{0}}%
\regfield{CPU\_GPIO\_OUT[1]}{1}{1}{{0}}%
\regfield{CPU\_GPIO\_OUT[0]}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:GPIO_OUT}\item [CPU\_GPIO\_OUT] Configures GPIOn (n=0 \verb+~+ 21) output value. It is a CPU CSR to write value (1=high, 0=low) to SoC GPIO pin. The value takes effect only when \hyperref[fielddesc:GPIO_OEN]{CPU\_GPIO\_OEN} is set.\\
CPU\_GPIO\_OUT[7:0] correspond to output signals cpu\_gpio\_out[7:0] in Table \ref{tab:iomuxgpio-periph-signals-via-hp-gpio-matrix-en} \textit{Peripheral Signals via GPIO Matrix}. \\
CPU\_GPIO\_OUT[7:0] can only be mapped to GPIO pins through GPIO matrix. For details please refer to Section \ref{sec:peripheral-output-via-gpio-matrix} in Chapter \textit{\nameref{mod:iomuxgpio}}.\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}
