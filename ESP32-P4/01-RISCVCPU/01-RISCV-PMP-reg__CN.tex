\begin{register}{H}{pmpXcfg}{}
\label{regdesc:pmpXcfg}%
\regfield{L}{1}{7}{{0}}%
\regfield{Reserved}{2}{5}{{0}}%
\regfield{A}{2}{3}{{0}}%
\regfield{X}{1}{2}{{0}}%
\regfield{W}{1}{1}{{0}}%
\regfield{R}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:pmpXcfgL}\item [L] 配置锁定位。一旦设置，只能通过核心复位来清除。\\
(R/W)
\label{fielddesc:pmpXcfgA}\item [A] 配置地址匹配模式。 \\
0：关闭 \\
1：TOR（范围顶端）\\
2：不支持 \\
3：NAPOT（自然对齐的 2 的幂次方区域，大小 >= 128 字节）\\
(R/W)
\label{fielddesc:pmpXcfgX}\item [X] 配置执行权限。设置后，允许从地址范围执行。(R/W)
\label{fielddesc:pmpXcfgW}\item [W] 配置写入权限。设置后，允许对地址范围进行存储器写入。(R/W)
\label{fielddesc:pmpXcfgR}\item [R] 配置读取权限。设置后，允许从地址范围进行存储器读取。(R/W)
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{pmpaddr\regindex{n} (\regindex{n}: 0-15)}{0x3B0+0x1*\regindex{n}}\label{regdesc:pmpaddrN}
\regfield{address}{32}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:ADDR}\item [address] 配置 pmpaddr\regindex{n} 的编码地址。(R/W)
\end{reglist}\end{regdesc}
\end{register}
