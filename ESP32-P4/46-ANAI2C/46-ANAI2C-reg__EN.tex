\begin{register}{H}{ANA\_I2C\_MST\_I2C0\_CTRL\_REG}{0x0000}\label{regdesc:ANAI2CMSTI2C0CTRLREG}
\regfield{(reserved)}{6}{26}{000000}%
\regfield{ANA\_I2C\_MST\_I2C0\_BUSY}{1}{25}{{0}}%
\regfield{ANA\_I2C\_MST\_I2C0\_CTRL}{25}{0}{{0x00000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:ANAI2CMSTI2C0CTRL}\item [ANA\_I2C\_MST\_I2C0\_CTRL] 
Configures the transmission information for I2C0. \\
Bit[0:7]: Configures the slave address\\
Bit[8:15]: Configures the slave register address\\
Bit[16:23]: Configures the transmitted data\\
Bit[24]: Configures the read or write operation\\
0: Write \\
1: Read \\
Once this register is configured, the I2C0 master will generate I2C read or write signals.\\
(R/W)
\label{fielddesc:ANAI2CMSTI2C0BUSY}\item [ANA\_I2C\_MST\_I2C0\_BUSY] Represents whether I2C0 is currently transferring data. \\
0: I2C0 is not transferring data\\
1: I2C0 is transferring data \\
(RO)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{ANA\_I2C\_MST\_ANA\_CONF2\_REG}{0x0020}\label{regdesc:ANAI2CMSTANACONF2REG}
\regfield{reserved}{8}{24}{{0x0}}%
\regfield{ANA\_I2C\_MST\_ANA\_CONF2}{24}{0}{{0x0000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:ANAI2CMSTANACONF2}\item [ANA\_I2C\_MST\_ANA\_CONF2] Configures which I2C master the following analog modules uses. If the corresponding bit is set to 1, I2C0 master is used for communication; if set to 0, I2C1 master is used. Since the slave address has already been defined, erroneous communication will not be caused.\\ 
Bit5: SYS\_PLL\\
Bit6: SDIO\_PLL\\
Bit7: ANA\_SENSOR\\
Bit8: PLLA\\
Bit9: MSPI\\
Bit10: DIG\_REG\\
Bit11: CPU\_PLL\\
Bit12: BIAS\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}
