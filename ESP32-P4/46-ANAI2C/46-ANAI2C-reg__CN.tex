\begin{register}{H}{ANA\_I2C\_MST\_I2C0\_CTRL\_REG}{0x0000}\label{regdesc:ANAI2CMSTI2C0CTRLREG}
\regfield{(reserved)}{6}{26}{000000}%
\regfield{ANA\_I2C\_MST\_I2C0\_BUSY}{1}{25}{{0}}%
\regfield{ANA\_I2C\_MST\_I2C0\_CTRL}{25}{0}{{0x00000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:ANAI2CMSTI2C0CTRL}\item [ANA\_I2C\_MST\_I2C0\_CTRL] 配置 I2C0 传输信息。\\
bit[0:7]：配置从机地址\\
bit[8:15]：配置从机寄存器地址\\
bit[16:23]：配置传输数据\\
bit[24]：配置读写操作\\
0：写\\
1：读\\
该寄存器配置完成后 I2C0 主机将产生 I2C 读写信号。\\
(R/W)
\label{fielddesc:ANAI2CMSTI2C0BUSY}\item [ANA\_I2C\_MST\_I2C0\_BUSY] 表示 I2C0 是否正在传输数据。\\
0：I2C0 未进行数据传输\\
1：I2C0 正在传输数据\\
(RO)
\end{reglist}\end{regdesc}
\end{register}
