\begin{register}{H}{XTS\_AES\_LINESIZE\_REG}{0x0340}\label{regdesc:XTSAESLINESIZEREG}

(R/W)

\regfield{XTS\_AES\_DESTINATION}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_DESTINATION] 配置手动加密的存储器类型，目前只能手动加密 flash，所以只能为 0。用户不能写入 1，否则将发生错误。\\

1：加密片外 RAM\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{XTS\_AES\_DPA\_CTRL\_REG}{0x0388}\label{regdesc:XTSAESDPACTRLREG}
\regfield{(reserved)}{27}{5}{{0x0000000}}%
\regfield{XTS\_AES\_CRYPT\_DPA\_SELECT\_REGISTER}{1}{4}{{0x0}}%
\regfield{XTS\_AES\_CRYPT\_CALC\_D\_DPA\_EN}{1}{3}{{0x1}}%
\regfield{XTS\_AES\_CRYPT\_SECURITY\_LEVEL}{3}{0}{{0x7}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:XTSAESCRYPTDPASELECTREGISTER}\item [XTS\_AES\_CRYPT\_DPA\_SELECT\_REGISTER] 配置由 eFuse 或寄存器控制抗 DPA 功能。\\
0：由寄存器控制抗 DPA 功能\\
1：由 eFuse 控制抗 DPA 功能\\
(R/W)
\label{fielddesc:XTSAESCRYPTCALCDDPAEN}\item [XTS\_AES\_CRYPT\_CALC\_D\_DPA\_EN] 配置是否在 XTS\_AES 算法中启用抗 DPA 功能。 \\
0：仅在计算 T 时启用抗 DPA 功能\\
1：计算 T 或是计算 D 时都启用抗 DPA 功能\\
请注意，此字段仅在 \hyperref[fielddesc:XTSAESCRYPTSECURITYLEVEL]{XTS\_AES\_CRYPT\_SECURITY\_LEVEL} 不为 0 时有效。\\
(R/W)
\label{fielddesc:XTSAESCRYPTSECURITYLEVEL}\item [XTS\_AES\_CRYPT\_SECURITY\_LEVEL] 配置外部加解密的安全等级。\\
0：抗 DPA 功能已关闭\\
1-7：数字越大，则加解密安全性更高\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{XTS\_AES\_STATE\_REG}{0x0358}\label{regdesc:XTSAESSTATEREG}
\regfield{(reserved)}{30}{2}{{0x00000000}}%
\regfield{XTS\_AES\_STATE}{2}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_STATE] 表示手动加密模块状态。\\
0 (XTS\_AES\_IDLE)：空闲\\
1 (XTS\_AES\_BUSY)：计算中\\
2 (XTS\_AES\_DONE)：计算完成，但手动加密结果数据对 SPI 不可见\\
3 (XTS\_AES\_RELEASE)：手动加密结果对 SPI 可见\\
(RO)
\end{reglist}\end{regdesc}
\end{register}
