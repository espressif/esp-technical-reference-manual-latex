\hypertarget{frac_3_4}{\textbf{3/4 编码方式}}
\begin{align*}
    BLOCKN_{i}^{j}[7:0]
        &= BLOCKN[48i + 8j + 7 : 48i + 8j]
        \qquad i \in \{0,1,2,3\} \qquad j \in \{0,1,2,3,4,5\} \\
    {}^{e}\!BLOCKN_{i}^{j}[7:0]
        &= {}^{e}\!BLOCKN[64i + 8j + 7 : 64i + 8j]
        \qquad i \in \{0,1,2,3\} \qquad j \in \{0,1,2,3,4,5,6,7\}
\end{align*}
\begin{gather*}
    \text{当 } i \in \{0,1,2,3\}, \\
    BLOCKN_{i}^{j}[7:0] = 
    \begin{cases}
      BLOCKN_{i}^{j}[7:0] & j \in \{0,1,2,3,4,5\} \\[0.8em]
      BLOCKN_{i}^{0}[7:0] \oplus BLOCKN_{i}^{1}[7:0] \\
      \quad BLOCKN_{i}^{2}[7:0] \oplus BLOCKN_{i}^{3}[7:0] \\
      \quad BLOCKN_{i}^{4}[7:0] \oplus BLOCKN_{i}^{5}[7:0] & j = 6 \\[0.8em]
      \sum\limits_{l=0}^{5} (l+1) \sum\limits_{k=0}^{7} BLOCKN_{i}^{l}[k], & j = 7
    \end{cases} \\
    \oplus \text{ 表示按位异或} \\
    \sum \text{和} + \text{表示加}
  \end{gather*}
