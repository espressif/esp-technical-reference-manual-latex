\begin{register}{H}{SYSCON\_SYSCLK\_CONF\_REG}{0x0000}\label{regdesc:SYSCONSYSCLKCONFREG}
    \regfield{(reserved)}{22}{10}{0000000000000000000000}%
    \regfield{SYSCON\_PRE\_DIV\_CNT}{10}{0}{{0x0}}%
    \reglabel{Reset}\regnewline%
    \begin{regdesc}\begin{reglist}
    \label{fielddesc:SYSCONPREDIVCNT}\item [SYSCON\_PRE\_DIV\_CNT] Configures the divider value of CPU\_CLK when the source of CPU\_CLK is XTL\_CLK or RC\_FAST\_CLK. The value range is 0x0 \verb+~+ 0x3FF. CPU\_CLK =  XTL\_CLK ( or RC\_FAST\_CLK) / (the value of this field +1). (R/W)
    \end{reglist}\end{regdesc}
    \end{register}
