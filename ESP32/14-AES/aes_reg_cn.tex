本小节括号中的地址均为相对于 AES 基地址的地址偏移量（相对地址），具体基地址见章节 \ref{mod:sysmem} \textit{\nameref{mod:sysmem}} 中的表 \ref{table:Peripheral_Address_Mapping} \textit{外设地址映射}。寄存器绝对地址见章节 \ref{sec:aes-reg-summ} \textit{\nameref{sec:aes-reg-summ}}。

\begin{register}{H}{AES\_START\_REG}{0x000}\label{regdesc:AESSTARTREG}
\regfield{(reserved)}{31}{1}{{0x00000000}}%
\regfield{AES\_START}{1}{0}{x}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [AES\_START] 写入 1 使能 AES 运算。（只写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{AES\_IDLE\_REG}{0x004}\label{regdesc:AESIDLEREG}
\regfield{(reserved)}{31}{1}{{0x00000000}}%
\regfield{AES\_IDLE}{1}{0}{1}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [AES\_IDLE] AES 空闲寄存器。AES 加速器运行时读出 0，空闲时读出 1。（只读）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{AES\_MODE\_REG}{0x008}\label{regdesc:AESMODEREG}
\regfield{(reserved)}{29}{3}{{0x00000000}}%
\regfield{AES\_MODE}{3}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [AES\_MODE] 选择 AES 加速器运算模式。详情请见表 \ref{table:AESOperation}。（读／写）
\end{reglist}\end{regdesc}
\end{register}
