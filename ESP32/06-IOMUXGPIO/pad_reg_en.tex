\begin{register}{H}{IO\_MUX\_PIN\_CTRL}{0x00}\label{regdesc:IOMUXPINCTRL}
\regfield{(reserved)}{20}{12}{{0x0}}%
\regfield{PIN\_CTRL\_CLK3}{4}{8}{{0x0}}%
\regfield{PIN\_CTRL\_CLK2}{4}{4}{{0x0}}%
\regfield{PIN\_CTRL\_CLK1}{4}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
{\bfseries If you want to output clock for I2S\regindex{0} (I2S\regindex{0}\_CLK) to:}\\
CLK\_OUT1, then set PIN\_CTRL[3:0] = 0x0;\\
CLK\_OUT2, then set PIN\_CTRL[3:0] = 0x0 and PIN\_CTRL[7:4] = 0x0;\\
CLK\_OUT3, then set PIN\_CTRL[3:0] = 0x0 and PIN\_CTRL[11:8] = 0x0.\\
{\bfseries If you want to output clock for I2S\regindex{1} (I2S\regindex{1}\_CLK) to:}\\
CLK\_OUT1, then set PIN\_CTRL[3:0] = 0xF;\\
CLK\_OUT2, then set PIN\_CTRL[3:0] = 0xF and PIN\_CTRL[7:4] = 0x0;\\
CLK\_OUT3, then set PIN\_CTRL[3:0] = 0xF and PIN\_CTRL[11:8] = 0x0. 

{\bfseries If you want to output clock for APLL to}\\
CLK\_OUT1, then set PIN\_CTRL[3:0] = 0x6;\\
CLK\_OUT2, then set PIN\_CTRL[3:0] = 0x6 and PIN\_CTRL[7:4] = 0x6;\\
CLK\_OUT3, then set PIN\_CTRL[3:0] = 0x6 and PIN\_CTRL[11:8] = 0x6. (R/W)
\end{regdesc}
\end{register}

\vspace{-2em}
\begin{tiplisting}
\vspace{-2em}
\begin{itemize}
    \item Only the above mentioned combinations of clock source (i.e. I2S\regindex{0/1}\_CLK, APLL clock) and clock output pins (i.e. CLK\_OUT1 \verb+~+ 3) are possible.
    \item The CLK\_OUT1 \verb+~+ 3 can be found in the \hyperref[table:iomux_pads]{IO MUX Pin Summary}.
\end{itemize}
\end{tiplisting}
 

\begin{register}{H}{IO\_MUX\_\textcolor{red}{\it{x}}\_REG (\textcolor{red}{\it{x}}: 0-19, 20$^1$, 21-23, 25-27, 32-39)}{See Table \ref{tab:iomux_reg_list_en} for the addresses}
\regfield{(reserved)}{17}{15}{00000000000000000}%
\regfield{MCU\_SEL}{3}{12}{{0x0}}%
\regfield{FUN\_DRV}{2}{10}{{0x2}}%
\regfield{FUN\_IE}{1}{9}{0}%
\regfield{FUN\_WPU}{1}{8}{0}%
\regfield{FUN\_WPD}{1}{7}{0}%
\regfield{MCU\_DRV}{2}{5}{{0x0}}%
\regfield{MCU\_IE}{1}{4}{0}%
\regfield{MCU\_WPU}{1}{3}{0}%
\regfield{MCU\_WPD}{1}{2}{0}%
\regfield{SLP\_SEL}{1}{1}{0}%
\regfield{MCU\_OE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [MCU\_SEL]Select the IO\_MUX function for this signal. 0 selects Function 0, 1 selects Function 1, etc. (R/W)
\label{FUNDRV}\item [FUN\_DRV]Select the drive strength of the pin. A higher value corresponds with a higher strength. For GPIO34-39, FUN\_DRV is always 0. For detailed drive strength, please see note 8 in Table "Notes on ESP32 Pin Lists", in \href{https://www.espressif.com/sites/default/files/documentation/esp32_datasheet_en.pdf}{\it{ESP32 Datasheet}}.  (R/W)
\item [FUN\_IE]Input enable of the pin. 1: input enabled; 0: input disabled. (R/W)
\label{regdesc:FUNWPU} \item  [FUN\_WPU]Pull-up enable of the pin. 1: internal pull-up enabled; 0: internal pull-up disabled. GPIO pins 34-39 are input-only. These pins do not feature an output driver or internal pull- up/pull-down circuitry, therefore, their FUN\_WPU is always 0. (R/W)
\label{regdesc:FUNWPD} \item  [FUN\_WPD]Pull-down enable of the pin. 1: internal pull-down enabled, 0: internal pull-down disabled. GPIO pins 34-39 are input-only. These pins do not feature an output driver or internal pull- up/pull-down circuitry, therefore, their FUN\_WPD is always 0. (R/W)
\item [MCU\_DRV]Select the drive strength of the pin during sleep mode. A higher value corresponds with a higher strength. (R/W)
\item [MCU\_IE]Input enable of the pin during sleep mode. 1: input enabled; 0: input disabled. (R/W)
\item [MCU\_WPU]Pull-up enable of the pin during sleep mode. 1: internal pull-up enabled; 0: internal pull-up disabled. (R/W)
\item [MCU\_WPD]Pull-down enable of the pin during sleep mode. 1: internal pull-down enabled; 0: internal pull-down disabled. (R/W)
\item [SLP\_SEL]Sleep mode selection of this pin. Set to 1 to put the pin in sleep mode. (R/W)
\item [MCU\_OE]Output enable of the pin in sleep mode. 1: enable output; 0: disable output. (R/W)
\end{reglist}\end{regdesc}

\end{tiplisting}
