\begin{register}{H}{APB\_SARADC\_CTRL\_REG}{0x10}\label{regdesc:APBSARADCCTRLREG}
\regfield{(reserved)}{5}{27}{00000}%
\regfield{APB\_SARADC\_DATA\_TO\_I2S}{1}{26}{0}%
\regfield{APB\_SARADC\_DATA\_SAR\_SEL}{1}{25}{0}%
\regfield{APB\_SARADC\_SAR2\_PATT\_P\_CLEAR}{1}{24}{0}%
\regfield{APB\_SARADC\_SAR1\_PATT\_P\_CLEAR}{1}{23}{0}%
\regfield{APB\_SARADC\_SAR2\_PATT\_LEN}{4}{19}{{15}}%
\regfield{APB\_SARADC\_SAR1\_PATT\_LEN}{4}{15}{{15}}%
\regfield{APB\_SARADC\_SAR\_CLK\_DIV}{8}{7}{{4}}%
\regfield{APB\_SARADC\_SAR\_CLK\_GATED}{1}{6}{1}%
\regfield{APB\_SARADC\_SAR\_SEL}{1}{5}{0}%
\regfield{APB\_SARADC\_WORK\_MODE}{2}{3}{{0}}%
\regfield{APB\_SARADC\_SAR2\_MUX}{1}{2}{0}%
\regfield{APB\_SARADC\_START}{1}{1}{0}%
\regfield{APB\_SARADC\_START\_FORCE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [APB\_SARADC\_DATA\_TO\_I2S] 1：I2S (DMA) 输入数据来自 SAR ADC；0：I2S 输入数据来自 GPIO 矩阵。（读/写）
\item [APB\_SARADC\_DATA\_SAR\_SEL] 1：sar\_sel 将由 16 位输出数据的 MSB 编码，此时分辨率不应大于 11 位；0：此时 SAR ADC 分辨率应为 12 位。（读/写）
\item [APB\_SARADC\_SAR2\_PATT\_P\_CLEAR] DIG ADC2 CTRL 的样式表指针清零。（读/写）
\item [APB\_SARADC\_SAR1\_PATT\_P\_CLEAR] DIG ADC1 CTRL 的样式表指针清零。（读/写）
\item [APB\_SARADC\_SAR2\_PATT\_LEN] SAR ADC2，0 - 15 代表样式表的 1 - 16 位。（读/写）
\item [APB\_SARADC\_SAR1\_PATT\_LEN] SAR ADC1，0 - 15 代表样式表的 1 - 16 位。（读/写）
\item [APB\_SARADC\_SAR\_CLK\_DIV] SAR 时钟分频器。（读/写）
\item [APB\_SARADC\_SAR\_CLK\_GATED] 保留位，请初始化为 0b1 。（读/写）
\item [APB\_SARADC\_SAR\_SEL] 0：SAR1；1：SAR2，该设置适用单通道 SAR 模式。（读/写）
\item [APB\_SARADC\_WORK\_MODE]0：单通道模式；1：双通道模式；2：交替模式。（读/写）
\item [APB\_SARADC\_SAR2\_MUX] 1：SAR ADC2 由 DIG ADC2 CTRL控制；0：SAR ADC2 由 PWDET CTRL 控制。（读/写）
\item [APB\_SARADC\_START] 保留位，请初始化为 0 。（读/写）
\item [APB\_SARADC\_START\_FORCE] 保留位，请初始化为 0 。（读/写）
\end{reglist}\end{regdesc}
\end{register}

\item [APB\_SARADC\_MEAS\_NUM\_LIMIT] 保留位，请初始化为 0b1 。（读/写）
\end{reglist}\end{regdesc}
\end{register}
