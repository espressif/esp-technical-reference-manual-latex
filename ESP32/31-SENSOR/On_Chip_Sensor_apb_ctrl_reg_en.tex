\begin{register}{H}{APB\_SARADC\_CTRL\_REG}{0x10}\label{regdesc:APBSARADCCTRLREG}
\regfield{(reserved)}{5}{27}{00000}%
\regfield{APB\_SARADC\_DATA\_TO\_I2S}{1}{26}{0}%
\regfield{APB\_SARADC\_DATA\_SAR\_SEL}{1}{25}{0}%
\regfield{APB\_SARADC\_SAR2\_PATT\_P\_CLEAR}{1}{24}{0}%
\regfield{APB\_SARADC\_SAR1\_PATT\_P\_CLEAR}{1}{23}{0}%
\regfield{APB\_SARADC\_SAR2\_PATT\_LEN}{4}{19}{{15}}%
\regfield{APB\_SARADC\_SAR1\_PATT\_LEN}{4}{15}{{15}}%
\regfield{APB\_SARADC\_SAR\_CLK\_DIV}{8}{7}{{4}}%
\regfield{APB\_SARADC\_SAR\_CLK\_GATED}{1}{6}{1}%
\regfield{APB\_SARADC\_SAR\_SEL}{1}{5}{0}%
\regfield{APB\_SARADC\_WORK\_MODE}{2}{3}{{0}}%
\regfield{APB\_SARADC\_SAR2\_MUX}{1}{2}{0}%
\regfield{APB\_SARADC\_START}{1}{1}{0}%
\regfield{APB\_SARADC\_START\_FORCE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [APB\_SARADC\_DATA\_TO\_I2S]1: I2S input data is from SAR ADC (for DMA), 0: I2S input data is from GPIO matrix. (R/W)
\item [APB\_SARADC\_DATA\_SAR\_SEL]1: sar\_sel will be coded by the MSB of the 16-bit output data, in this case, the resolution should not contain more than 11 bits; 0: using 12-bit SAR ADC resolution. (R/W)
\item [APB\_SARADC\_SAR2\_PATT\_P\_CLEAR]Clears the pointer of pattern table for DIG ADC2 CTRL. (R/W)
\item [APB\_SARADC\_SAR1\_PATT\_P\_CLEAR]Clears the pointer of pattern table for DIG ADC1 CTRL. (R/W)
\item [APB\_SARADC\_SAR2\_PATT\_LEN]SAR ADC2, 0 - 15 means pattern table length of 1 - 16. (R/W)
\item [APB\_SARADC\_SAR1\_PATT\_LEN]SAR ADC1, 0 - 15 means pattern table length of 1 - 16. (R/W)
\item [APB\_SARADC\_SAR\_CLK\_DIV]SAR clock divider. (R/W)
\item [APB\_SARADC\_SAR\_CLK\_GATED]Reserved. Please initialize to 0b1 (R/W)
\item [APB\_SARADC\_SAR\_SEL]0: SAR1, 1: SAR2, this setting is applicable in the single SAR mode. (R/W)
\item [APB\_SARADC\_WORK\_MODE]0: single mode, 1: double mode, 2: alternate mode. (R/W)
\item [APB\_SARADC\_SAR2\_MUX]1: SAR ADC2 is controlled by DIG ADC2 CTRL, 0: SAR ADC2 is controlled by PWDET CTRL. (R/W)
\item [APB\_SARADC\_START]Reserved. Please initialize to 0 (R/W)
\item [APB\_SARADC\_START\_FORCE]Reserved. Please initialize to 0 (R/W)
\end{reglist}\end{regdesc}
\end{register}

\end{reglist}\end{regdesc}
\end{register}
