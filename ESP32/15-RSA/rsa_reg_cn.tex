本小节括号中的地址均为相对于 RSA 基地址的地址偏移量（相对地址），具体基地址见章节 \ref{mod:sysmem} \textit{\nameref{mod:sysmem}} 中的表 \ref{table:Peripheral_Address_Mapping} \textit{外设地址映射}。寄存器绝对地址见章节 \ref{sec:rsa-reg-summ} \textit{\nameref{sec:rsa-reg-summ}}。


\begin{register}{H}{RSA\_M\_PRIME\_REG}{0x800}\label{regdesc:RSAMPRIMEREG}
\regfield{}{32}{0}{{0x000000000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RSA\_M\_PRIME\_REG] 此寄存器包含 M'。（读/写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{RSA\_MODEXP\_MODE\_REG}{0x804}\label{regdesc:RSAMODEXPMODEREG}
\regfield{(reserved)}{29}{3}{00000000000000000000000000000}%
\regfield{RSA\_MODEXP\_MODE}{3}{0}{000}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RSA\_MODEXP\_MODE] 此寄存器包含模幂运算的模式。（读/写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{RSA\_MODEXP\_START\_REG}{0x808}\label{regdesc:RSAMODEXPSTARTREG}
\regfield{(reserved)}{31}{1}{0000000000000000000000000000000}%
\regfield{RSA\_MODEXP\_START}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RSA\_MODEXP\_START] 写入 1 以开始模幂运算。（只写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{RSA\_MULT\_MODE\_REG}{0x80C}\label{regdesc:RSAMULTMODEREG}
\regfield{(reserved)}{28}{4}{0000000000000000000000000000}%
\regfield{RSA\_MULT\_MODE}{4}{0}{0000}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RSA\_MULT\_MODE] 此寄存器包含模乘和乘法运算。（只写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{RSA\_MULT\_START\_REG}{0x810}\label{regdesc:RSAMULTSTARTREG}
\regfield{(reserved)}{31}{1}{0000000000000000000000000000000}%
\regfield{RSA\_MULT\_START}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RSA\_MULT\_START] 写入 1 以开始模乘和乘法运算。（只写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{RSA\_INTERRUPT\_REG}{0x814}\label{regdesc:RSAINTERRUPTREG}
\regfield{(reserved)}{31}{1}{0000000000000000000000000000000}%
\regfield{RSA\_INTERRUPT}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RSA\_INTERRUPT] RSA 中断状态寄存器。一次运算结束即读到 1。（读/写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{RSA\_CLEAN\_REG}{0x818}\label{regdesc:RSACLEANREG}
\regfield{(reserved)}{31}{1}{0000000000000000000000000000000}%
\regfield{RSA\_CLEAN}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RSA\_CLEAN] 一旦存储器初始化结束，此位即读到 1。（只读）
\end{reglist}\end{regdesc}
\end{register}
