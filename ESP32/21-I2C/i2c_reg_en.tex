\begin{register}{H}{I2C\_CTR\_REG}{0x0004}\label{regdesc:I2CCTRREG}
\regfield{(reserved)}{24}{8}{000000000000000000000000}%
\regfield{I2C\_RX\_LSB\_FIRST}{1}{7}{0}%
\regfield{I2C\_TX\_LSB\_FIRST}{1}{6}{0}%
\regfield{I2C\_TRANS\_START}{1}{5}{0}%
\regfield{I2C\_MS\_MODE}{1}{4}{0}%
\regfield{(reserved)}{1}{3}{0}%
\regfield{I2C\_SAMPLE\_SCL\_LEVEL}{1}{2}{0}%
\regfield{I2C\_SCL\_FORCE\_OUT}{1}{1}{1}%
\regfield{I2C\_SDA\_FORCE\_OUT}{1}{0}{1}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [I2C\_RX\_LSB\_FIRST]This bit is used to control the storage mode for received data. (R/W) \\1: receive data from the least significant bit;    \\0: receive data from the most significant bit.
\item [I2C\_TX\_LSB\_FIRST]This bit is used to control the sending mode for data needing to be sent. (R/W) \\1: send data from the least significant bit;    \\0: send data from the most significant bit.
\item [I2C\_TRANS\_START]Set this bit to start sending the data in txfifo. (R/W)
\item [I2C\_MS\_MODE]Set this bit to configure the module as an I2C Master. Clear this bit to configure the module as an I2C Slave. (R/W)
\item [I2C\_SAMPLE\_SCL\_LEVEL] 1: sample SDA data on the SCL low level; 0: sample SDA data on the SCL high level. (R/W)
\item [I2C\_SCL\_FORCE\_OUT] 0: direct output; 1: open drain output. (R/W)
\item [I2C\_SDA\_FORCE\_OUT] 0: direct output; 1: open drain output. (R/W)
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{I2C\_SR\_REG}{0x0008}\label{regdesc:I2CSRREG}
\regfield{(reserved)}{1}{31}{0}%
\regfield{I2C\_SCL\_STATE\_LAST}{3}{28}{000}%
\regfield{(reserved)}{1}{27}{0}%
\regfield{I2C\_SCL\_MAIN\_STATE\_LAST}{3}{24}{000}%
\regfield{I2C\_TXFIFO\_CNT}{6}{18}{000000}%
\regfield{(reserved)}{4}{14}{0000}%
\regfield{I2C\_RXFIFO\_CNT}{6}{8}{000000}%
\regfield{(reserved)}{1}{7}{0}%
\regfield{I2C\_BYTE\_TRANS}{1}{6}{0}%
\regfield{I2C\_SLAVE\_ADDRESSED}{1}{5}{0}%
\regfield{I2C\_BUS\_BUSY}{1}{4}{0}%
\regfield{I2C\_ARB\_LOST}{1}{3}{0}%
\regfield{I2C\_TIME\_OUT}{1}{2}{0}%
\regfield{I2C\_SLAVE\_RW}{1}{1}{0}%
\regfield{I2C\_ACK\_REC}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [I2C\_SCL\_STATE\_LAST] This field  indicates the states of the state machine used to produce SCL. (RO)\\
0: Idle; 1: Start; 2: Negative edge; 3: Low; 4: Positive edge; 5: High; 6: Stop
\item [I2C\_SCL\_MAIN\_STATE\_LAST] This field indicates the states of the I2C module state machine. (RO)\\
0: Idle; 1: Address shift; 2: ACK address; 3: Rx data; 4: Tx data; 5: Send ACK; 6: Wait ACK
\item [I2C\_TXFIFO\_CNT]This field stores the amount of received data  in RAM. (RO)
\item [I2C\_RXFIFO\_CNT]This field represents the amount of data needed to be sent. (RO)
\item [I2C\_BYTE\_TRANS]This field changes to 1 when one byte is transferred. (RO)
\item [I2C\_SLAVE\_ADDRESSED]When configured as an I2C Slave, and the address sent by the master is equal to the address of the slave, then this bit will be of high level. (RO)
\item [I2C\_BUS\_BUSY]1: the I2C bus is busy transferring data; 0: the I2C bus is in idle state. (RO)
\item [I2C\_ARB\_LOST]When the I2C controller loses control of SCL line, this register changes to 1. (RO)
\item [I2C\_TIME\_OUT]When the I2C controller takes more than I2C\_TIME\_OUT clocks to receive a data bit, this field changes to 1. (RO)
\item [I2C\_SLAVE\_RW]When in slave mode, 1: master reads from slave; 0: master writes to slave. (RO)
\item [I2C\_ACK\_REC]This register stores the value of the received ACK bit. (RO)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{I2C\_FIFO\_CONF\_REG}{0x0018}\label{regdesc:I2CFIFOCONFREG}
\regfield{(reserved)}{6}{26}{000000}%
\regfield{I2C\_NONFIFO\_TX\_THRES}{6}{20}{{0x15}}%
\regfield{I2C\_NONFIFO\_RX\_THRES}{6}{14}{{0x15}}%
\regfield{(reserved)}{2}{12}{00}%
\regfield{I2C\_FIFO\_ADDR\_CFG\_EN}{1}{11}{0}%
\regfield{I2C\_NONFIFO\_EN}{1}{10}{0}%
\regfield{I2C\_TXFIFO\_EMPTY\_THRHD}{5}{5}{0}%
\regfield{I2C\_RXFIFO\_FULL\_THRHD}{5}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [I2C\_NONFIFO\_TX\_THRES]When I2C sends more than nonfifo\_tx\_thres bytes of data, it will generate a tx\_send\_empty\_int\_raw interrupt and update the current offset address of the sent data. (R/W)
\item [I2C\_NONFIFO\_RX\_THRES]When I2C receives more than nonfifo\_rx\_thres bytes of data, it will generate a rx\_send\_full\_int\_raw interrupt and update the current offset address of the received data. (R/W)
\item [I2C\_FIFO\_ADDR\_CFG\_EN]When this bit is set to 1, the byte received after the I2C address byte represents the offset address in the I2C Slave RAM. (R/W)
\item [I2C\_NONFIFO\_EN]Set this bit to enable APB nonfifo access. (R/W)
\item [I2C\_TXFIFO\_EMPTY\_THRHD] Configures the TX FIFO threshold in non-FIFO mode. When the TX FIFO count is greater than I2C\_TXFIFO\_EMPTY\_THRHD [4:0], I2C\_TXFIFO\_EMPTY\_INT\_RAW bit is valid. (R/W)
\item [I2C\_RXFIFO\_FULL\_THRHD] Configures the RX FIFO threshold in non-FIFO mode. When the RX FIFO count is greater than I2C\_RXFIFO\_FULL\_THRHD [4:0], I2C\_RXFIFO\_FULL\_INT\_RAW bit is valid. (R/W)
\end{reglist}\end{regdesc}
\end{register}
