\begin{register}{H}{RMT\_CH\textit{\textcolor{red}{\it{n}}}CONF0\_REG (\textcolor{red}{\it{n}}: 0-7)}{0x0020+8*\textcolor{red}{\it{n}}}\label{regdesc:RMTCHnCONF0REG}
\regfield{(reserved)}{1}{31}{{0x0}}%
\regfield{RMT\_MEM\_PD}{1}{30}{0}%
\regfield{RMT\_CARRIER\_OUT\_LV\_CH\textcolor{red}{\it{n}}}{1}{29}{1}%
\regfield{RMT\_CARRIER\_EN\_CH\textcolor{red}{\it{n}}}{1}{28}{1}%
\regfield{RMT\_MEM\_SIZE\_CH\textcolor{red}{\it{n}}}{4}{24}{{0x01}}%
\regfield{RMT\_IDLE\_THRES\_CH\textcolor{red}{\it{n}}}{16}{8}{{0x01000}}%
\regfield{RMT\_DIV\_CNT\_CH\textcolor{red}{\it{n}}}{8}{0}{{0x002}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [RMT\_MEM\_PD]This bit is used to power down the entire RMT RAM block. (It only exists in RMT\_CH0CONF0). 1: power down memory; 0: power up memory. (R/W)
\item [RMT\_CARRIER\_OUT\_LV\_CH\textcolor{red}{\it{n}}]This bit is used for configuration when the carrier wave is being transmitted. Transmit on low output level with 0, and transmit on high output level with 1. (R/W)
\item [RMT\_CARRIER\_EN\_CH\textcolor{red}{\it{n}}]This is the carrier modulation enable-control bit for channel \textcolor{red}{\it{n}}. Carrier modulation is enabled with 1, while carrier modulation is disabled with 0. (R/W)
\item [RMT\_MEM\_SIZE\_CH\textcolor{red}{\it{n}}]This register is used to configure the amount of memory blocks allocated to channel \textcolor{red}{\it{n}}. (R/W)
\item [RMT\_IDLE\_THRES\_CH\textcolor{red}{\it{n}}]In receive mode, when no edge is detected on the input signal for longer than REG\_IDLE\_THRES\_CH\textcolor{red}{\it{n}} channel clock cycles, the receive process is finished. (R/W)
\item [RMT\_DIV\_CNT\_CH\textcolor{red}{\it{n}}]This register is used to set the divider for the channel clock of channel \textcolor{red}{\it{n}}. (R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{regdesc}\begin{reglist}
\item [RMT\_IDLE\_OUT\_EN\_CH\textcolor{red}{\it{n}}]This is the output enable-control bit for channel \textcolor{red}{\it{n}} in IDLE state. (R/W)
\item [RMT\_IDLE\_OUT\_LV\_CH\textcolor{red}{\it{n}}]This bit configures the level of output signals in channel \textcolor{red}{\it{n}} when the latter is in IDLE state. (R/W)
\item [RMT\_REF\_ALWAYS\_ON\_CH\textcolor{red}{\it{n}}]This bit is used to select the channel's base clock. 1:clk\_apb; 0:clk\_ref. (R/W)
\item [RMT\_REF\_CNT\_RST\_CH\textcolor{red}{\it{n}}]Setting this bit resets the clock divider of channel \textcolor{red}{\it{n}}. (R/W)
\item [RMT\_RX\_FILTER\_THRES\_CH\textcolor{red}{\it{n}}]In receive mode, channel \textcolor{red}{\it{n}} ignores input pulse when the pulse width is smaller than this value in APB clock periods. (R/W)
\item [RMT\_RX\_FILTER\_EN\_CH\textcolor{red}{\it{n}}]This is the receive filter's enable-bit for channel \textcolor{red}{\it{n}}. (R/W)
\item [RMT\_TX\_CONTI\_MODE\_CH\textcolor{red}{\it{n}}]If this bit is set, instead of going to an idle state when transmission ends, the transmitter will restart transmission. This results in a repeating output signal. (R/W)
\item [RMT\_MEM\_OWNER\_CH\textcolor{red}{\it{n}}]This bit marks channel \textcolor{red}{\it{n}}'s RAM block ownership. Number 1 indicates that the receiver is using the RAM, while 0 indicates that the transmitter is using the RAM. (R/W)
\item [RMT\_MEM\_RD\_RST\_CH\textcolor{red}{\it{n}}]Set this bit to reset the read-RAM address for channel \textcolor{red}{\it{n}} by accessing the transmitter. (R/W)
\item [RMT\_MEM\_WR\_RST\_CH\textcolor{red}{\it{n}}]Set this bit to reset the write-RAM address for channel \textcolor{red}{\it{n}} by accessing the receiver. (R/W)
\item [RMT\_RX\_EN\_CH\textcolor{red}{\it{n}}]Set this bit to enable receiving data on channel \textcolor{red}{\it{n}}. (R/W)
\item [RMT\_TX\_START\_CH\textcolor{red}{\it{n}}]Set this bit to start sending data on channel \textcolor{red}{\it{n}}. (R/W)
\end{reglist}\end{regdesc}
\end{register}
