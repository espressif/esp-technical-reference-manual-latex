\begin{register}{H}{FLASH\_ENCRYPTION\_BUFFER\_\textcolor{red}{\it{n}}\_REG (\textcolor{red}{\it{n}}: 0-7)}{0x0+4*\textcolor{red}{\it{n}}}\label{regdesc:FLASHENCRYPTIONBUFFERnREG}

\begin{register}{H}{FLASH\_ENCRYPTION\_START\_REG}{0x020}\label{regdesc:FLASHENCRYPTIONSTARTREG}

\regfield{FLASH\_START}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [FLASH\_START] 将此位置为 1，启动加密操作。（只写）
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{FLASH\_ENCRYPTION\_ADDRESS\_REG}{0x024}\label{regdesc:FLASHENCRYPTIONADDRESSREG}
\regfield{}{32}{0}{{0x000000000}}%

\begin{regdesc}\begin{reglist}

\end{reglist}\end{regdesc}
\end{register}
