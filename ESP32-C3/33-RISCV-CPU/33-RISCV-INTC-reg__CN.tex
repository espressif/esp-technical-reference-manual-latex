
\begin{register}{H}{INT\_ENABLE\_REG}{0x0104}\label{regdesc:INTENABLEREG}
\regfield{INT\_ENABLE}{31}{1}{{0x00000000}}%
\regfield{(reserved)}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\item [INT\_ENABLE\lbrack\regindex{n}\rbrack] 置位第 \regindex{n} 个位使能中断 \regindex{n}。（读/写）
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0：禁能
    \item 1：使能
\end{itemize}
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{INT\_TYPE\_REG}{0x0108}\label{regdesc:INTTYPEREG}
\regfield{INT\_TYPE}{31}{1}{{0x00000000}}%
\regfield{(reserved)}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\item [INT\_TYPE\lbrack\regindex{n}\rbrack] 置位第 \regindex{n} 个位捕获中断  \regindex{n} 的上升沿。（读/写）
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0：电平类型（检测信号电平）
    \item 1：脉冲类型（检测上升沿）
\end{itemize}
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{INT\_CLEAR\_REG}{0x010C}\label{regdesc:INTCLEARREG}
\regfield{INT\_CLEAR}{31}{1}{{0x00000000}}%
\regfield{(reserved)}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\item [INT\_CLEAR\lbrack\regindex{n}\rbrack] 置位第 \regindex{n} 个位清除中断  \regindex{n} 的等待状态。（读/写） \newline 仅对“脉冲”类型的中断有效，“电平”类型中断应在中断源进行清除。 \newline 注意置位后需要手动切换至 0。
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0：无关项
    \item 1：清除等待状态
\end{itemize}
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{INT\_EIP\_REG}{0x0110}\label{regdesc:INTEIPREG}
\regfield{INT\_EIP}{31}{1}{{0x00000000}}%
\regfield{(reserved)}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\item [INT\_EIP\lbrack\regindex{n}\rbrack] 读取第 \regindex{n} 个位获得中断  \regindex{n} 的等待状态。（只读） \newline 只有被使能的且阈值以上的中断才会反映在该域中。
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0：不在等待
    \item 1：在等待
\end{itemize}
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{INT\_PRIORITY\_\regindex{n}\_REG (\regindex{n}: 1-31)}{0x0114+4*\regindex{n}}\label{regdesc:INTPRIORITYREG}
\regfield{(reserved)}{28}{4}{{0x00000000}}%
\regfield{INT\_PRIORITY\_\regindex{n}}{4}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:INTPRIORITY}\item [INT\_PRIORITY\_\regindex{n}] 写入 4 位数值到第 \regindex{n} 个寄存器配置中断 \regindex{n} 的优先级。（读/写） \newline \textit{说明：不管阈值是多少，优先级为 0 的中断都会被屏蔽。}
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{INT\_THRESH\_REG}{0x0194}\label{regdesc:INTTHRESHREG}
\regfield{(reserved)}{28}{4}{{0x00000000}}%
\regfield{INT\_THRESH}{4}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:INTTHRESH}\item [INT\_THRESH] 写入 4 位的数值配置所有中断的全局优先级阈值。（读/写） \newline 所有优先级低于阈值的中断都会被屏蔽。 \newline \textit{说明：不管阈值是多少，优先级为 0 的中断都会被屏蔽。}
\end{reglist}
\end{regdesc}
\end{register}
