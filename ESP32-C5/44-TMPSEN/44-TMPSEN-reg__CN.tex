\begin{register}{H}{APB\_SARADC\_APB\_TSENS\_CTRL2\_REG}{0x005C}\label{regdesc:APBSARADCAPBTSENSCTRL2REG}
\regfield{(reserved)}{16}{16}{0000000000000000}%
\regfield{APB\_SARADC\_TSENS\_CLK\_SEL}{1}{15}{{0}}%
\regfield{APB\_SARADC\_TSENS\_CLK\_INV}{1}{14}{{0x0}}%
\regfield{APB\_SARADC\_TSENS\_XPD\_FORCE}{2}{12}{{0x0}}%
\regfield{APB\_SARADC\_TSENS\_XPD\_WAIT}{12}{0}{{0x2}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:APBSARADCTSENSCLKSEL}\item [APB\_SARADC\_TSENS\_CLK\_SEL] 配置温度传感器时钟。\\
0：RC\_FAST\_CLK\\
1：XTAL\_CLK\\
(R/W)
\label{fielddesc:APBSARADCTSENSCLKINV}\item [APB\_SARADC\_TSENS\_CLK\_INV] 配置传感器采样时钟的相位。(R/W)
\label{fielddesc:APBSARADCTSENSXPDFORCE}\item [APB\_SARADC\_TSENS\_XPD\_FORCE] 配置是否使能强制上电/断电温度传感器。\\
0：禁用强制上电功能\\
1：禁用强制断电功能\\
2：强制上电温度传感器\\
3：强制断电温度传感器\\
(R/W)
\label{fielddesc:APBSARADCTSENSXPDWAIT}\item [APB\_SARADC\_TSENS\_XPD\_WAIT] 配置温度传感器的复位释放前需要等待的时钟周期。(R/W)
\end{reglist}\end{regdesc}
\end{register}
