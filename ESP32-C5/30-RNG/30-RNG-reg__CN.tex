
\begin{register}{H}{LPPERI\_RNG\_CFG\_REG}{0x0024}\label{regdesc:LPPERIRNGCFGREG}
\regfield{(reserved)}{8}{24}{{0x0}}%
\regfield{(reserved)}{12}{12}{000000000000}%
\regfield{LPPERI\_RTC\_TIMER\_EN}{2}{10}{{3}}%
\regfield{(reserved)}{9}{1}{000000000}%
\regfield{LPPERI\_RNG\_SAMPLE\_ENABLE}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:LPPERIRNGSAMPLEENABLE}\item [LPPERI\_RNG\_SAMPLE\_ENABLE] 配置是否使能 BUF\_CHAIN。\\0：关闭\\ 1：使能\\  (R/W)
\label{fielddesc:LPPERIRTCTIMEREN}\item [LPPERI\_RTC\_TIMER\_EN]
Bit[0]：配置是否在 CRC 前使能 RTC 定时器。\\
Bit[1]:配置是否在 CRC 后使能 RTC 定时器。\\
0：关闭\\ 1：使能\\ (R/W)
\end{reglist}\end{regdesc}
\end{register}
