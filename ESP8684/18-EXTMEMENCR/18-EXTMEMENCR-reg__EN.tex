\begin{register}{H}{XTS\_AES\_LINESIZE\_REG}{0x0040}\label{regdesc:XTSAESLINESIZEREG}
\regfield{(reserved)}{31}{1}{{0x00000000}}%
\regfield{XTS\_AES\_LINESIZE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_LINESIZE] Configures the data size of one encryption operation. (R/W)
    \begin{itemize}
        \item 0: 16 bytes;
        \item 1: 32 bytes.
    \end{itemize}
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{XTS\_AES\_DESTINATION\_REG}{0x0044}\label{regdesc:XTSAESDESTINATIONREG}
\regfield{(reserved)}{31}{1}{{0x00000000}}%
\regfield{XTS\_AES\_DESTINATION}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_DESTINATION] Configures the type of the external memory. Currently, it must be set to 0,
as the Manual Encryption block only supports flash encryption. Errors may occur if users write 1. (R/W)
    \begin{itemize}
        \item 0: flash;
        \item 1: external RAM.
    \end{itemize}

\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{XTS\_AES\_STATE\_REG}{0x0058}\label{regdesc:XTSAESSTATEREG}
\regfield{(reserved)}{30}{2}{{0x00000000}}%
\regfield{XTS\_AES\_STATE}{2}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_STATE] Indicates the status of the Manual Encryption block. (RO)
    \begin{itemize}
        \item 0x0 (XTS\_AES\_IDLE): idle;
        \item 0x1 (XTS\_AES\_BUSY): busy with encryption;
        \item 0x2 (XTS\_AES\_DONE): encryption is completed, but the encrypted result is not accessible to SPI;
        \item 0x3 (XTS\_AES\_RELEASE): encrypted result is accessible to SPI.
    \end{itemize}
\end{reglist}\end{regdesc}
\end{register}
