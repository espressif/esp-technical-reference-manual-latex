\begin{register}{H}{mcause}{0x342}\label{regdesc:MCAUSE}
\regfield{Interrupt Flag}{1}{31}{{0}}%
\regfield{(reserved)}{26}{5}{{0x0000000}}%
\regfield{Exception Code}{5}{0}{{0x00}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:MCAUSECODE}\item [Exception Code] This field is automatically updated with unique ID of the most recent exception or interrupt due to which CPU entered trap. (R/W)
\newline Possible exception IDs are:
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0x1: PMP Instruction access fault
    \item 0x2: Illegal Instruction
    \item 0x3: Hardware Breakpoint/Watchpoint or EBREAK
    \item 0x5: PMP Load access fault
    \item 0x7: PMP Store access fault
    \item 0x8: ECALL from U mode
    \item 0xb: ECALL from M mode
\end{itemize}
\textit {Note: Exception ID 0x0 (instruction access misaligned) is not present because CPU always masks the lowest bit of the address during instruction fetch.}
\label{fielddesc:MCAUSEINT}\item [Interrupt Flag] This flag is automatically updated when CPU enters trap. (R/W) \newline If this is found to be set, indicates that the latest trap occurred due to interrupt. For exceptions it remains unset. \newline \textit {Note: The interrupt controller is using up IDs in range 1-31 for all external interrupt sources. This is different from the RISC-V standard which has reserved IDs in range 0-15 for core internal interrupt sources.}
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{mtval}{0x343}\label{regdesc:MTVAL}

\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:MTVAL}\item [MTVAL] Machine trap value. (R/W) \newline This is automatically updated with an exception dependent data which may be useful for handling that exception.
\newline Data is to be interpreted depending upon exception IDs:
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0x1: Faulting virtual address of instruction
    \item 0x2: Faulting instruction opcode
    \item 0x5: Faulting data address of load operation
    \item 0x7: Faulting data address of store operation
\end{itemize}
\textit {Note: The value of this register is not valid for  other exception IDs and interrupts.}
\end{reglist}\end{regdesc}
