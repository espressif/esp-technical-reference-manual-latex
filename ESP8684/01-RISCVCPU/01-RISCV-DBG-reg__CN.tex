
\begin{register}{H}{dcsr}{0x7B0}
\label{regdesc:DCSR}
\regfield{xdebugver}{4}{28}{{4}}%
\regfield{reserved}{12}{16}{{0}}%
\regfield{ebreakm}{1}{15}{{0}}%
\regfield{reserved}{2}{13}{{0}}%
\regfield{ebreaku}{1}{12}{{0}}%
\regfield{reserved}{1}{11}{{0}}%
\regfield{stopcount}{1}{10}{{0}}%
\regfield{stoptime}{1}{9}{{0}}%
\regfield{cause}{3}{6}{{0}}%
\regfield{reserved}{3}{3}{{0}}%
\regfield{step}{1}{2}{{0}}%
\regfield{prv}{2}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:DCSRXEDBGVER}\item [xdebugver] 调试版本。（只读）
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 4：存在外部调试支持
\end{itemize}
\label{fielddesc:DCSREBREAKM}\item [ebreakm] 置位后，机器模式中的 ebreak 指令进入调试模式。（读/写）
\label{fielddesc:DCSREBREAKU}\item [ebreaku] 置位后，用户/程序模式中的 ebreak 指令进入调试模式。（读/写）
\label{fielddesc:DCSRSTOPCOUNT}\item [stopcount] 此域没有实现。调试器会始终读出 0。（只读）
\label{fielddesc:DCSRSTOPTIME}\item [stoptime] 此性能没有实现。调试器会始终读出 0。（只读）
\label{fielddesc:DCSRCAUSE}\item [cause] 说明进入调试模式的原因。当单个周期中有多个原因导致进入调试模式，会反映出具有最高优先级数值的那个原因。（只读）
\begin{enumerate}
    \setlength\itemsep{-1em}
    \item 执行了一条 ebreak 指令（优先级 3）
    \item 触发模块引起暂停（优先级 4）
    \item haltreq 被置位（优先级 2）
    \item step 被置位导致 CPU 单步执行（优先级 1）
\end{enumerate}
其他值保留供以后使用。
\label{fielddesc:DCSRSTEP}\item [step] 当被置位且不处于调试模式时，内核将仅执行单个指令，然后进入调试模式。当该位置 1 时，中断被\textbf{使能}*。如果指令由于异常而未能完成，则内核将在执行异常处理程序之前立即进入调试模式，并置位相应的异常寄存器。（读/写）
\label{fielddesc:DCSRPRV}\item [prv] 保存 CPU 进入调试模式时候的特权级别。退出调试模式时，调试器可以更改此值以改变内核的特权级别。仅支持 \textbf{0x3}（机器模式）和 \textbf{0x0}（用户模式）。
\end{reglist}
\end{regdesc}
\end{register}
*\textbf{注意}：与 RISC-V 调试规格版本 0.13 不同。

\begin{register}{H}{dpc}{0x7B1}
\label{regdesc:DPC}
\regfield{dpc}{32}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:DPC}\item [dpc] 进入调试模式后，dpc 将写入遇到异常的指令的虚拟地址。恢复执行时，CPU 内核的 PC 将更新为 dpc 保存的虚拟地址。调试器可以写入 dpc 配置 CPU 恢复执行的位置。（读/写）
\end{reglist}
\end{regdesc}
\end{register}
