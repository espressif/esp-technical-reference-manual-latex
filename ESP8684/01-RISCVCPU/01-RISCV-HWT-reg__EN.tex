
\begin{register}{H}{tselect}{0x7A0}\label{regdesc:TSELECT}
\regfield{(reserved)}{29}{3}{{0x00000000}}%
\regfield{tselect}{3}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:TSELECT}\item [tselect] Index (0-7) of the selected trigger unit. (R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{tdata1}{0x7A1}\label{regdesc:TDATA1}
\regfield{type}{4}{28}{{0x2}}%
\regfield{dmode}{1}{27}{{0}}%
\regfield{data}{27}{0}{{0x3e00000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:TDATA1TYPE}\item [type] Type of trigger. (RO) \newline This field is reserved since only match type (0x2) triggers are supported.
\label{fielddesc:TDATA1DMODE}\item [dmode] This is set to 1 if a trigger is being used by the debugger. (R/W *)
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0: Both Debug and machine-mode can write the tdata1 and tdata2
registers at the selected tselect.
    \item 1: Only Debug Mode can write the tdata1 and tdata2 registers at the selected tselect. Writes from other modes are ignored.
\end{itemize}
\textit{* Note : Only writable from debug mode.}
\label{fielddesc:TDATA1DATA}\item [data] Abstract tdata1 content. (R/W) \newline This will always be interpreted as fields of \hyperref[regdesc:MCONTROL]{mcontrol} since only match type (0x2) triggers are supported.
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{tdata2}{0x7A2}\label{regdesc:TDATA2}
\regfield{tdata2}{32}{0}{{0x00000000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:TDATA2}\item [tdata2] Abstract tdata2 content. (R/W) \newline This will always be interpreted as \hyperref[regdesc:MADDRESS]{maddress} since only match type (0x2) triggers are supported.
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{tcontrol}{0x7A5}\label{regdesc:TCONTROL}
\regfield{(reserved)}{24}{8}{{0x000000}}%
\regfield{mpte}{1}{7}{{0}}%
\regfield{(reserved)}{6}{1}{{0x00}}%
\regfield{mte}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:TCONTROLMPTE}\item [mpte] Machine-mode previous trigger enable bit. (R/W)
\begin{itemize}
    \setlength\itemsep{-1em}
    \item When CPU is taking a machine-mode trap, the value of \hyperref[fielddesc:TCONTROLMTE]{mte} is automatically pushed into this.
    \item When CPU is executing MRET, its value is popped back into \hyperref[fielddesc:TCONTROLMTE]{mte}, so this becomes 0.
\end{itemize}
\label{fielddesc:TCONTROLMTE}\item [mte] Machine-mode trigger enable bit. (R/W)
\begin{itemize}
    \setlength\itemsep{-1em}
    \item When CPU is taking a machine-mode trap, its value is automatically pushed into \hyperref[fielddesc:TCONTROLMPTE]{mpte}, so this becomes 0 and triggers with \hyperref[fielddesc:MCONTROLACTION]{action}=0 are disabled globally.
    \item When CPU is executing MRET, the value of \hyperref[fielddesc:TCONTROLMPTE]{mpte} is automatically popped back into this.
\end{itemize}
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{mcontrol}{0x7A1}\label{regdesc:MCONTROL}
\regfield{(reserved)}{4}{28}{{0x2}}%
\regfield{dmode}{1}{27}{{0}}%
\regfield{(reserved)}{6}{21}{{0x1f}}%
\regfield{hit}{1}{20}{{0}}%
\regfield{(reserved)}{4}{16}{{0}}%
\regfield{action}{4}{12}{{0}}%
\regfield{(reserved)}{1}{11}{{0}}%
\regfield{match}{4}{7}{{0}}%
\regfield{m}{1}{6}{{0}}%
\regfield{(reserved)}{2}{4}{{0}}%
\regfield{u}{1}{3}{{0}}%
\regfield{execute}{1}{2}{{0}}%
\regfield{store}{1}{1}{{0}}%
\regfield{load}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:MCONTROLDMODE}\item [dmode] Same as \hyperref[fielddesc:TDATA1DMODE]{dmode} in \hyperref[regdesc:TDATA1]{tdata1}.
\label{fielddesc:MCONTROLHIT}\item [hit] This is found to be 1 if the selected trigger had fired previously. (R/W) \newline This bit is to be cleared manually.
\label{fielddesc:MCONTROLACTION}\item [action] Write this for configuring the selected trigger to perform one of the available actions when firing. (R/W)
\newline Valid options are:
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0x0: cause breakpoint exception.
    \item 0x1: enter debug mode (only valid when dmode = 1)
\end{itemize}
\textit{Note : Writing an invalid value will set this to the default value 0x0.}
\label{fielddesc:MCONTROLMATCH}\item [match] Write this for configuring the selected trigger to perform one of the available matching operations on a data/instruction address. (R/W)
Valid options are:
\begin{itemize}
    \setlength\itemsep{-1em}
    \item 0x0: exact byte match, i.e., address corresponding to one of the bytes in an access must match the value of \hyperref[regdesc:MADDRESS]{maddress} exactly.
    \item 0x1: NAPOT match, i.e., at least one of the bytes of an access must lie in the NAPOT region specified in \hyperref[regdesc:MADDRESS]{maddress}.
\end{itemize}
\textit{Note : Writing a larger value will clip it to the largest possible value 0x1.}
\label{fielddesc:MCONTROLM}\item [m] Set this for enabling selected trigger to operate in machine-mode. (R/W)
\label{fielddesc:MCONTROLU}\item [u] Set this for enabling selected trigger to operate in user-mode. (R/W)
\label{fielddesc:MCONTROLEXECUTE}\item [execute] Set this for configuring the selected trigger to fire right before an instruction with matching virtual address is executed by the CPU. (R/W)
\label{fielddesc:MCONTROLSTORE}\item [store] Set this for configuring the selected trigger to fire right before a store operation with matching data address is executed by the CPU. (R/W)
\label{fielddesc:MCONTROLLOAD}\item [load] Set this for configuring the selected trigger to fire right before a load operation with matching data address is executed by the CPU. (R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{maddress}{0x7A2}\label{regdesc:MADDRESS}
\regfield{maddress}{32}{0}{{0x00000000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:MADDRESS}\item [maddress] Address used by the selected trigger when performing match operation. (R/W) \newline This is decoded as NAPOT when \hyperref[fielddesc:MCONTROLMATCH]{match}=1 in \hyperref[regdesc:MCONTROL]{mcontrol}.
\end{reglist}\end{regdesc}
\end{register}
