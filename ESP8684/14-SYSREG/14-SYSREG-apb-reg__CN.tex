\begin{register}{H}{SYSCON\_CLKGATE\_FORCE\_ON\_REG}{0x00A4}\label{regdesc:SYSCONCLKGATEFORCEONREG}
\regfield{(reserved)}{25}{7}{00000000000000000000000000}%
\regfield{SYSCON\_SRAM\_CLKGATE\_FORCE\_ON}{4}{3}{{0xf}}%
\regfield{SYSCON\_ROM\_CLKGATE\_FORCE\_ON}{3}{0}{{7}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONROMCLKGATEFORCEON}\item [SYSCON\_ROM\_CLKGATE\_FORCE\_ON] 置 1 配置 ROM 内存的时钟门控始终打开；置 0 则配置 ROM 内存的时钟门控在被访问时自动打开，没有访问时自动关闭。(R/W)
\label{fielddesc:SYSCONSRAMCLKGATEFORCEON}\item [SYSCON\_SRAM\_CLKGATE\_FORCE\_ON] 置 1 配置 SRAM 内存的时钟门控始终打开；置 0 则配置 SRAM 内存的时钟门控在被访问时自动打开，没有访问时自动关闭。(R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{SYSCON\_MEM\_POWER\_DOWN\_REG}{0x00A8}\label{regdesc:SYSCONMEMPOWERDOWNREG}
\regfield{(reserved)}{25}{7}{00000000000000000000000000}%
\regfield{SYSCON\_SRAM\_POWER\_DOWN}{4}{3}{{0}}%
\regfield{SYSCON\_ROM\_POWER\_DOWN}{3}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONROMPOWERDOWN}\item [SYSCON\_ROM\_POWER\_DOWN] 控制 Internal ROM 进入 Retention 状态。(R/W)
\label{fielddesc:SYSCONSRAMPOWERDOWN}\item [SYSCON\_SRAM\_POWER\_DOWN] 控制 Internal SRAM 进入 Retention 状态。(R/W)
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{SYSCON\_MEM\_POWER\_UP\_REG}{0x00AC}\label{regdesc:SYSCONMEMPOWERUPREG}
\regfield{(reserved)}{25}{7}{00000000000000000000000000}%
\regfield{SYSCON\_SRAM\_POWER\_UP}{4}{3}{{0xf}}%
\regfield{SYSCON\_ROM\_POWER\_UP}{3}{0}{{7}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONROMPOWERUP}\item [SYSCON\_ROM\_POWER\_UP] 控制 Internal ROM 在芯片进入 Light-sleep 时不会进入 Retention 状态。(R/W)
\label{fielddesc:SYSCONSRAMPOWERUP}\item [SYSCON\_SRAM\_POWER\_UP] 控制 Internal SRAM 在芯片进入 Light-sleep 时不会进入 Retention 状态。(R/W)
\end{reglist}\end{regdesc}
\end{register}
