\begin{register}{H}{SYSCON\_CLKGATE\_FORCE\_ON\_REG}{0x00A4}\label{regdesc:SYSCONCLKGATEFORCEONREG}
\regfield{(reserved)}{25}{7}{00000000000000000000000000}%
\regfield{SYSCON\_SRAM\_CLKGATE\_FORCE\_ON}{4}{3}{{0xf}}%
\regfield{SYSCON\_ROM\_CLKGATE\_FORCE\_ON}{3}{0}{{7}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONROMCLKGATEFORCEON}\item [SYSCON\_ROM\_CLKGATE\_FORCE\_ON] Set this field to configure the ROM clock gate to be always on; Set 0 to configure the clock gate to turn on automatically when ROM is accessed and turn off automatically when ROM is not accessed. (R/W)
\label{fielddesc:SYSCONSRAMCLKGATEFORCEON}\item [SYSCON\_SRAM\_CLKGATE\_FORCE\_ON] Set this field to configure the SRAM clock gate to be always on; Set 0 to configure the clock gate to turn on automatically when SRAM is accessed and turn off automatically when SRAM is not accessed. (R/W)
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{SYSCON\_MEM\_POWER\_DOWN\_REG}{0x00A8}\label{regdesc:SYSCONMEMPOWERDOWNREG}
\regfield{(reserved)}{25}{7}{00000000000000000000000000}%
\regfield{SYSCON\_SRAM\_POWER\_DOWN}{4}{3}{{0}}%
\regfield{SYSCON\_ROM\_POWER\_DOWN}{3}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONROMPOWERDOWN}\item [SYSCON\_ROM\_POWER\_DOWN] Set this field to send the internal ROM into retention state. (R/W)
\label{fielddesc:SYSCONSRAMPOWERDOWN}\item [SYSCON\_SRAM\_POWER\_DOWN] Set this field to send the internal SRAM into retention state. (R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{SYSCON\_MEM\_POWER\_UP\_REG}{0x00AC}\label{regdesc:SYSCONMEMPOWERUPREG}
\regfield{(reserved)}{25}{7}{00000000000000000000000000}%
\regfield{SYSCON\_SRAM\_POWER\_UP}{4}{3}{{0xf}}%
\regfield{SYSCON\_ROM\_POWER\_UP}{3}{0}{{7}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONROMPOWERUP}\item [SYSCON\_ROM\_POWER\_UP] Set this field to force the internal ROM to work as normal (do not enter the retention state) when the chip enters light sleep. (R/W)
\label{fielddesc:SYSCONSRAMPOWERUP}\item [SYSCON\_SRAM\_POWER\_UP] Set this field to force the internal SRAM to work as normal (do not enter the retention state) when the chip enters light sleep. (R/W)
\end{reglist}\end{regdesc}
\end{register}
