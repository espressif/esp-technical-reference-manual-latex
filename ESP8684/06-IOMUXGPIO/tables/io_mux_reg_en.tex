\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXGPIONMCUOE}\item [IO\_MUX\_GPIO\regindex{n}\_MCU\_OE] Output enable of the pin in sleep mode. 1: output enabled; 0: output disabled. (R/W)
\label{fielddesc:IOMUXGPIONSLPSEL}\item [IO\_MUX\_GPIO\regindex{n}\_SLP\_SEL] Sleep mode selection of this pin. Set to 1 to put the pin in sleep mode. (R/W)
\label{fielddesc:IOMUXGPIONMCUWPD}\item [IO\_MUX\_GPIO\regindex{n}\_MCU\_WPD] Pull-down enable of the pin in sleep mode. 1: internal pull-down enabled; 0: internal pull-down disabled. (R/W)
\label{fielddesc:IOMUXGPIONMCUWPU}\item [IO\_MUX\_GPIO\regindex{n}\_MCU\_WPU] Pull-up enable of the pin during sleep mode. 1: internal pull-up enabled; 0: internal pull-up disabled. (R/W)
\label{fielddesc:IOMUXGPIONMCUIE}\item [IO\_MUX\_GPIO\regindex{n}\_MCU\_IE] Input enable of the pin during sleep mode. 1: input enabled; 0: input disabled. (R/W)
\label{fielddesc:IOMUXGPIONMCUDRV}\item [IO\_MUX\_GPIO\regindex{n}\_MCU\_DRV] Configures the drive strength of GPIO\regindex{n} during sleep mode. \\
0: \verb+~+5 mA\\
1: \verb+~+ 10 mA\\
2: \verb+~+ 20 mA\\
3: \verb+~+40 mA\\ (R/W)
\label{fielddesc:IOMUXGPIONFUNWPD}\item [IO\_MUX\_GPIO\regindex{n}\_FUN\_WPD] Pull-down enable of the pin. 1: internal pull-down enabled; 0: internal pull-down disabled. (R/W)
\label{fielddesc:IOMUXGPIONFUNWPU}\item [IO\_MUX\_GPIO\regindex{n}\_FUN\_WPU] Pull-up enable of the pin. 1: internal pull-up enabled; 0: internal pull-up disabled. (R/W)
\label{fielddesc:IOMUXGPIONFUNIE}\item [IO\_MUX\_GPIO\regindex{n}\_FUN\_IE] Input enable of the pin. 1: input enabled; 0: input disabled. (R/W)
\label{fielddesc:IOMUXGPIONFUNDRV}\item [IO\_MUX\_GPIO\regindex{n}\_FUN\_DRV] Select the drive strength of the pin. 0: \verb+~+ 5 mA; 1: \verb+~+ 10 mA; 2: \verb+~+ 20 mA; 3: \verb+~+ 40 mA. (R/W)
\label{fielddesc:IOMUXGPIONMCUSEL}\item [IO\_MUX\_GPIO\regindex{n}\_MCU\_SEL] Select IO MUX function for this pin. 0: Select Function 0; 1: Select Function 1; etc. (R/W)
\label{fielddesc:IOMUXGPIONFILTEREN}\item [IO\_MUX\_GPIO\regindex{n}\_FILTER\_EN] Enable filter for pin input signals. 1: filter enabled; 0: filter disabled. (R/W)
\end{reglist}\end{regdesc}
\end{register}
