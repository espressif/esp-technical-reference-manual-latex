\begin{register}{H}{IO\_MUX\_PIN\_CTRL\_REG}{\hexprefix{}0000}\label{regdesc:IOMUXPINCTRL}
\regfield{(reserved)}{16}{16}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_PAD\_POWER\_CTRL}{1}{15}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_SWITCH\_PRT\_NUM}{3}{12}{{\hexprefix{}2}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK3}{4}{8}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK2}{4}{4}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK1}{4}{0}{{\hexprefix{}0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXPINCTRLCLK1}\item [IO\_MUX\_PIN\_CTRL\_CLK1] Configure I2S0 clock output: (R/W)
\begin{itemize}
    \item 0: output I2S0 clock to CLK\_OUT1.
    \item 15: disabled.
\end{itemize}
\label{fielddesc:IOMUXPINCTRLCLK2}\item [IO\_MUX\_PIN\_CTRL\_CLK2] Configure I2S0 clock output: (R/W)
\begin{itemize}
    \item 0: output I2S0 clock to CLK\_OUT2.
    \item 15: disabled.
\end{itemize}
\label{fielddesc:IOMUXPINCTRLCLK3}\item [IO\_MUX\_PIN\_CTRL\_CLK3] Configure I2S0 clock output: (R/W)
\begin{itemize}
    \item 0: output I2S0 clock to CLK\_OUT3.
    \item 15: disabled.
\end{itemize}


\begin{tiplisting}
Only the above mentioned combinations of clock source and clock output pins are possible. \\
The CLK\_OUT1 \verb+~+ 3 can be found in \hyperref[table:iomux_pads]{IO\_MUX Pad List}.
\end{tiplisting}
\item [IO\_MUX\_SWITCH\_PRT\_NUM] IO pin power switch delay, delay unit is one APB clock. (R/W)
\item [IO\_MUX\_PAD\_POWER\_CTRL] Select power voltage for GPIO33 \verb+~+ 37. 1: select VDD\_SPI 1.8 V; 0: select VDD3P3\_CPU 3.3 V. (R/W)

\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{IO\_MUX\_\regindex{n}\_REG (\regindex{n}: GPIO0-GPIO21, GPIO26-GPIO46)}{\hexprefix{}0004+4*\regindex{n}}\label{regdesc:IOMUXnREG} 
\regfield{(reserved)}{16}{16}{0000000000000000}%
\regfield{IO\_MUX\_FILTER\_EN}{1}{15}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_MCU\_SEL}{3}{12}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_FUN\_DRV}{2}{10}{{\hexprefix{}2}}%
\regfield{IO\_MUX\_FUN\_IE}{1}{9}{0}%
\regfield{IO\_MUX\_FUN\_WPU}{1}{8}{0}%
\regfield{IO\_MUX\_FUN\_WPD}{1}{7}{0}%
\regfield{(reserved)}{2}{5}{{00}}%
\regfield{IO\_MUX\_MCU\_IE}{1}{4}{0}%
\regfield{IO\_MUX\_MCU\_WPU}{1}{3}{0}%
\regfield{IO\_MUX\_MCU\_WPD}{1}{2}{0}%
\regfield{IO\_MUX\_SLP\_SEL}{1}{1}{0}%
\regfield{IO\_MUX\_MCU\_OE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXMCUOE}\item [IO\_MUX\_MCU\_OE] Output enable of the pin in sleep mode. 1: Output enabled; 0: Output disabled. (R/W)
\label{fielddesc:IOMUXSLPSEL}\item [IO\_MUX\_SLP\_SEL] Sleep mode selection of this pin. Set to 1 to put the pin in sleep mode. (R/W)
\label{fielddesc:IOMUXMCUWPD}\item [IO\_MUX\_MCU\_WPD] Pull-down enable of the pin during sleep mode. 1: Internal pull-down enabled; 0: Internal pull-down disabled. (R/W)
\label{fielddesc:IOMUXMCUWPU}\item [IO\_MUX\_MCU\_WPU] Pull-up enable of the pin during sleep mode. 1: Internal pull-up enabled; 0: Internal pull-up disabled.
\label{fielddesc:IOMUXMCUIE}\item [IO\_MUX\_MCU\_IE] Input enable of the pin during sleep mode. 1: Input enabled; 0: Input disabled. (R/W)
\label{fielddesc:IOMUXFUNWPD}\item [IO\_MUX\_FUN\_WPD] Pull-down enable of the pin. 1: Pull-down enabled; 0: Pull-down disabled. (R/W)
\label{fielddesc:IOMUXFUNWPU}\item [IO\_MUX\_FUN\_WPU] Pull-up enable of the pin. 1: Internal pull-up enabled; 0: Internal pull-up disabled. (R/W)
\label{fielddesc:IOMUXFUNIE}\item [IO\_MUX\_FUN\_IE] Input enable of the pin. 1: Input enabled; 0: Input disabled. (R/W)
\label{fielddesc:IOMUXFUNDRV}\item [IO\_MUX\_FUN\_DRV] Select the drive strength of the pin. 0: \verb+~+5 mA: 1: \verb+~+10 mA: 2: \verb+~+20 mA; 3: \verb+~+40 mA. (R/W)
\label{fielddesc:IOMUXMCUSEL}\item [IO\_MUX\_MCU\_SEL] Select IO MUX function for this signal. 0: Select Function 0; 1: Select Function 1, etc. (R/W)
\label{fielddesc:IOMUXFILTEREN}\item [IO\_MUX\_FILTER\_EN] Enable filter for pin input signals. 1: Filter enabled; 0: Filter disabled. (R/W)

\end{reglist}\end{regdesc}
\end{register}
