\begin{register}{H}{IO\_MUX\_PIN\_CTRL\_REG}{\hexprefix{}0000}\label{regdesc:IOMUXPINCTRL}
\regfield{(reserved)}{16}{16}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_PAD\_POWER\_CTRL}{1}{15}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_SWITCH\_PRT\_NUM}{3}{12}{{\hexprefix{}2}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK3}{4}{8}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK2}{4}{4}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_PIN\_CTRL\_CLK1}{4}{0}{{\hexprefix{}0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXPINCTRLCLK1}\item [IO\_MUX\_PIN\_CTRL\_CLK1] 配置 I2S0 外设时钟输出。当 IO\_MUX\_PIN\_CTRL1 为 0 时，输出 I2S0 时钟到 CLK\_OUT1；当 IO\_MUX\_PIN\_CTRL1 为 15 时，不输出时钟。(R/W)

\label{fielddesc:IOMUXPINCTRLCLK2}\item [IO\_MUX\_PIN\_CTRL\_CLK2] 配置 I2S0 外设时钟输出。当 IO\_MUX\_PIN\_CTRL2 为 0 时，输出 I2S0 时钟到 CLK\_OUT2；当 IO\_MUX\_PIN\_CTRL2 为 15 时，不输出时钟。(R/W)


\label{fielddesc:IOMUXPINCTRLCLK3}\item [IO\_MUX\_PIN\_CTRL\_CLK3] 配置 I2S0 外设时钟输出。当 IO\_MUX\_PIN\_CTRL3 为 0 时，输出 I2S0 时钟到 CLK\_OUT3；当 IO\_MUX\_PIN\_CTRL3 为 15 时，不输出时钟。(R/W)
\begin{tiplisting}
只能有上述三种配置组合。CLK\_OUT1 \verb+~+ 3 可在 \hyperref[table:iomux_pads]{IO\_MUX Pad 列表}中查询。
\end{tiplisting}
\label{fielddesc:IOMUXSWITCHPRTNUM}\item [IO\_MUX\_SWITCH\_PRT\_NUM] IO Pad电源切换延时，延时单位为一个 APB 时钟周期。(R/W)
\label{fielddesc:IOMUXPADPOWERCTRL}\item [IO\_MUX\_PAD\_POWER\_CTRL] 选择 GPIO33 \verb+~+ 37 的电源电压。1：选择 VDD\_SPI 1.8 V 供电；0：选择 VDD3P3\_CPU 3.3 V 供电。(R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{IO\_MUX\_\regindex{n}\_REG (\regindex{n}: GPIO0-GPIO21, GPIO26-GPIO46)}{\hexprefix{}0004+4*\regindex{n}}\label{regdesc:IOMUXnREG} 
\regfield{(reserved)}{16}{16}{0000000000000000}%
\regfield{IO\_MUX\_FILTER\_EN}{1}{15}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_MCU\_SEL}{3}{12}{{\hexprefix{}0}}%
\regfield{IO\_MUX\_FUN\_DRV}{2}{10}{{\hexprefix{}2}}%
\regfield{IO\_MUX\_FUN\_IE}{1}{9}{0}%
\regfield{IO\_MUX\_FUN\_WPU}{1}{8}{0}%
\regfield{IO\_MUX\_FUN\_WPD}{1}{7}{0}%
\regfield{(reserved)}{2}{5}{{00}}%
\regfield{IO\_MUX\_MCU\_IE}{1}{4}{0}%
\regfield{IO\_MUX\_MCU\_WPU}{1}{3}{0}%
\regfield{IO\_MUX\_MCU\_WPD}{1}{2}{0}%
\regfield{IO\_MUX\_SLP\_SEL}{1}{1}{0}%
\regfield{IO\_MUX\_MCU\_OE}{1}{0}{0}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:IOMUXMCUOE}\item [IO\_MUX\_MCU\_OE]睡眠模式下 pad 的输出使能。1：输出使能；0：输出关闭。（读/写）
\label{fielddesc:IOMUXSLPSEL}\item [IO\_MUX\_SLP\_SEL] Pad 的睡眠模式选择。置 1 将使能睡眠模式。（读/写）
\label{fielddesc:IOMUXMCUWPD}\item [IO\_MUX\_MCU\_WPD]睡眠模式下 pad 的下拉使能。1：内部下拉使能；0：内部下拉关闭。（读/写）
\label{fielddesc:IOMUXMCUWPU}\item [IO\_MUX\_MCU\_WPU]睡眠模式下 pad 的上拉使能。1：内部上拉使能；0：内部上拉关闭。（读/写）
\label{fielddesc:IOMUXMCUIE}\item [IO\_MUX\_MCU\_IE]睡眠模式下 pad 的输入使能。1：输入使能；0：输入关闭。（读/写）
\label{fielddesc:IOMUXFUNWPD}\item [IO\_MUX\_FUN\_WPD]Pad 的下拉使能。1：内部下拉使能；0：内部下拉关闭。（读/写）
\label{fielddesc:IOMUXFUNWPU}\item [IO\_MUX\_FUN\_WPU]Pad 的上拉使能。1：内部上拉使能；0：内部上拉关闭。（读/写）
\label{fielddesc:IOMUXFUNIE}\item [IO\_MUX\_FUN\_IE] Pad 的输入使能。1：输入使能；0：输入关闭。（读/写）
\label{fielddesc:IOMUXFUNDRV}\item [IO\_MUX\_FUN\_DRV]选择 Pad 驱动强度。0：\verb+~+5 mA；1：\verb+~+10 mA；2：\verb+~+20 mA；3：\verb+~+40 mA。（读/写）
\label{fielddesc:IOMUXMCUSEL}\item [IO\_MUX\_MCU\_SEL]为信号选择 IO MUX 功能。0：选择 Function 0；1：选择 Function 1；以此类推。（读/写）
\label{fielddesc:IOMUXFILTEREN}\item [IO\_MUX\_FILTER\_EN] Pad 输入信号滤波使能。1：滤波使能；0：滤波关闭。（读/写）

\end{reglist}\end{regdesc}
\end{register}
