\regfield{SENS\_SAR\_I2C\_START\_FORCE}{1}{29}{{0}}%
\regfield{SENS\_SAR\_I2C\_START}{1}{28}{{0}}%
\regfield{SENS\_SAR\_I2C\_WR\_EN}{1}{27}{{0}}%
\regfield{SENS\_SAR\_I2C\_WDATA}{8}{19}{{0}}%
\regfield{SENS\_SAR\_I2C\_REG\_ADDR}{8}{11}{{0}}%
\regfield{SENS\_SAR\_I2C\_SLAVE\_ADDR\_10BIT\_EN}{1}{10}{{0}}%

\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SENSSARI2CSLAVEADDR}\item [SENS\_SAR\_I2C\_SLAVE\_ADDR] 配置从机地址。（读/写）
\label{fielddesc:SENSSARI2CSLAVEADDR10BITEN}\item [SENS\_SAR\_I2C\_SLAVE\_ADDR\_10BIT\_EN] 配置是否将从机地址扩展为 10 位。\\
0：不扩展\\
1：扩展\\
（读/写）
\label{fielddesc:SENSSARI2CREGADDR}\item [SENS\_SAR\_I2C\_REG\_ADDR] 配置寄存器地址。（读/写）
\label{fielddesc:SENSSARI2CWDATA}\item [SENS\_SAR\_I2C\_WDATA] 配置写入的数据。（读/写）
\label{fielddesc:SENSSARI2CWREN}\item [SENS\_SAR\_I2C\_WR\_EN] 配置是否使用写命令。\\
0：使用读命令\\
1：使用写命令\\
（读/写）
\label{fielddesc:SENSSARI2CSTART}\item [SENS\_SAR\_I2C\_START] 启动 RTC I2C，仅当 SENS\_SAR\_I2C\_START\_FORCE = 1 时有效。（读/写）
\label{fielddesc:SENSSARI2CSTARTFORCE}\item [SENS\_SAR\_I2C\_START\_FORCE] RTC I2C 启动模式。0：由 FSM 启动；1：由软件启动。（读/写）
\end{reglist}\end{regdesc}
\end{register}
