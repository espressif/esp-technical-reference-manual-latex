\begin{register}{H}{XTS\_AES\_STATE\_REG}{\hexprefix{}0158}\label{regdesc:XTSAESSTATEREG}

\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:XTSAESSTATE}\item [XTS\_AES\_STATE] Indicates the status of the Manual Encryption block. (RO)
\begin{itemize}
    \item \hexprefix{}0 (XTS\_AES\_IDLE): idle;
    \item \hexprefix{}1 (XTS\_AES\_BUSY): busy with encryption;
    \item \hexprefix{}2 (XTS\_AES\_DONE): encryption is completed, but the encrypted result is not accessible to SPI;
    \item \hexprefix{}3 (XTS\_AES\_RELEASE): encrypted result is accessible to SPI.
\end{itemize}
