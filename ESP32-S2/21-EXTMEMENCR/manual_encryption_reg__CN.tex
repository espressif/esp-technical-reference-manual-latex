\begin{register}{H}{XTS\_AES\_LINESIZE\_REG}{\hexprefix{}0140}\label{regdesc:XTSAESLINESIZEREG}

\end{register}

\begin{register}{H}{XTS\_AES\_STATE\_REG}{\hexprefix{}0158}\label{regdesc:XTSAESSTATEREG}
\regfield{(reserved)}{30}{2}{{\hexprefix{}00000000}}%
\regfield{XTS\_AES\_STATE}{2}{0}{{\hexprefix{}0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [XTS\_AES\_STATE] 手动加密模块状态寄存器。（只读）
    \begin{itemize}
        \item \hexprefix{}0 (XTS\_AES\_IDLE)：空闲；
        \item \hexprefix{}1 (XTS\_AES\_BUSY)：忙于计算；
        \item \hexprefix{}2 (XTS\_AES\_DONE)：计算完成，但手动加密结果数据对 SPI 不可见；
        \item \hexprefix{}3 (XTS\_AES\_RELEASE)：手动加密结果对 SPI 可见。
    \end{itemize}
\end{reglist}\end{regdesc}
\end{register}
