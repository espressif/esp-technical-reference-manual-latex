\begin{register}{H}{SYSCON\_TICK\_CONF\_REG}{\hexprefix{}0004}\label{regdesc:SYSCONTICKCONFREG}
\regfield{(reserved)}{15}{17}{000000000000000}%
\regfield{SYSCON\_TICK\_ENABLE}{1}{16}{1}%
\regfield{SYSCON\_CK8M\_TICK\_NUM}{8}{8}{7}%
\regfield{SYSCON\_XTAL\_TICK\_NUM}{8}{0}{{39}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONXTALTICKNUM}\item [SYSCON\_XTAL\_TICK\_NUM] Configures the divider value of REF\_TICK when the source of CPU\_CLK is PLL\_CLK, XTAL\_CLK or APLL\_CLK. The value range is \hexprefix{}0 \verb+~+ \hexprefix{}FF. REF\_TICK = APB\_CLK /(the value of this field + 1). (R/W)
\label{fielddesc:SYSCONCK8MTICKNUM}\item [SYSCON\_CK8M\_TICK\_NUM] Configures the divider value of REF\_TICK when the source of CPU\_CLK is RC\_FAST\_CLK. The value range is \hexprefix{}0 \verb+~+ \hexprefix{}FF. REF\_TICK = APB\_CLK/(the value of this field + 1). (R/W)
\label{fielddesc:SYSCONXTALTICKENABLE}\item [SYSCON\_TICK\_ENABLE] Enable REF\_TICK. 1: Enable. 0: Disable. (R/W)
\end{reglist}\end{regdesc}
\end{register}
