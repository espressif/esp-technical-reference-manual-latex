\begin{register}{H}{SYSCON\_TICK\_CONF\_REG}{\hexprefix{}0004}\label{regdesc:SYSCONTICKCONFREG}
\regfield{(reserved)}{15}{17}{000000000000000}%
\regfield{SYSCON\_TICK\_ENABLE}{1}{16}{1}%
\regfield{SYSCON\_CK8M\_TICK\_NUM}{8}{8}{7}%
\regfield{SYSCON\_XTAL\_TICK\_NUM}{8}{0}{{39}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:SYSCONXTALTICKNUM}\item [SYSCON\_XTAL\_TICK\_NUM] 当 CPU\_CLK 的时钟源为 PLL\_CLK、XTAL\_CLK 或 APLL\_CLK 时，配置 REF\_TICK 的分频系数。取值范围为 \hexprefix{}0 \verb+~+ \hexprefix{}FF。REF\_TICK = APB\_CLK/(取值 + 1)。 (R/W)
\label{fielddesc:SYSCONCK8MTICKNUM}\item [SYSCON\_CK8M\_TICK\_NUM] 当 CPU\_CLK 的时钟源为 RC\_FAST\_CLK 时，配置 REF\_TICK 的分频系数。取值范围为 \hexprefix{}0 \verb+~+ \hexprefix{}FF。REF\_TICK = APB\_CLK/(取值 + 1)。 (R/W)
\label{fielddesc:SYSCONXTALTICKENABLE}\item [SYSCON\_TICK\_ENABLE] 使能 REF\_TICK。1：使能；0：关闭。(R/W)
\end{reglist}\end{regdesc}
\end{register}
