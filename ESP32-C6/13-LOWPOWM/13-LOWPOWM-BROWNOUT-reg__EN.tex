\begin{register}{H}{LP\_ANA\_BOD\_MODE0\_CNTL\_REG}{0x0000}\label{regdesc:LPANABODMODE0CNTLREG}\regfield{LP\_ANA\_BOD\_MODE0\_RESET\_ENA}{1}{31}{{0}}%
\regfield{LP\_ANA\_BOD\_MODE0\_RESET\_SEL}{1}{30}{{0}}%
\regfield{LP\_ANA\_BOD\_MODE0\_INTR\_ENA}{1}{29}{{0}}%
\regfield{LP\_ANA\_BOD\_MODE0\_CNT\_CLR}{1}{28}{{0}}%
\regfield{LP\_ANA\_BOD\_MODE0\_RESET\_WAIT}{10}{18}{{0x3ff}}%
\regfield{LP\_ANA\_BOD\_MODE0\_INTR\_WAIT}{10}{8}{{1}}%
\regfield{LP\_ANA\_BOD\_MODE0\_PD\_RF\_ENA}{1}{7}{{0}}%
\regfield{LP\_ANA\_BOD\_MODE0\_CLOSE\_FLASH\_ENA}{1}{6}{{0}}%
\regfield{(reserved)}{6}{0}{000000}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:LPANABODMODE0CLOSEFLASHENA}\item [LP\_ANA\_BOD\_MODE0\_CLOSE\_FLASH\_ENA] Configures whether to enable the brownout detector to trigger flash suspend. \\
0: Disable \\
1: Enable\\
(R/W)
\label{fielddesc:LPANABODMODE0PDRFENA}\item [LP\_ANA\_BOD\_MODE0\_PD\_RF\_ENA] Configures whether to enable the brownout detector to close the RF module. \\
0: Disable \\
1: Enable\\
(R/W)
\label{fielddesc:LPANABODMODE0INTRWAIT}\item [LP\_ANA\_BOD\_MODE0\_INTR\_WAIT] Configures the time to generate an interrupt after the brownout signal is valid. The unit is LP\_FAST\_CLK. (R/W)
\label{fielddesc:LPANABODMODE0RESETWAIT}\item [LP\_ANA\_BOD\_MODE0\_RESET\_WAIT] Configures the time to generate a reset after the brownout signal is valid. The unit is LP\_FAST\_CLK. (R/W)
\label{fielddesc:LPANABODMODE0CNTCLR}\item [LP\_ANA\_BOD\_MODE0\_CNT\_CLR] Configures whether to clear the count value of the brownout detector.\\
0: Do not clear\\
1: Clear\\
(R/W)
\label{fielddesc:LPANABODMODE0INTRENA}\item [LP\_ANA\_BOD\_MODE0\_INTR\_ENA] Enables the interrupts for the brownout detector mode 0. \hyperref[fielddesc:LPANABODMODE0INTRAW]{LP\_ANA\_BOD\_MODE0\_INT\_RAW} and \hyperref[fielddesc:LPANABODMODE0LPINTRAW]{LP\_ANA\_BOD\_MODE0\_LP\_INT\_RAW} are valid only when this field is set to 1. (R/W)
\label{fielddesc:LPANABODMODE0RESETSEL}\item [LP\_ANA\_BOD\_MODE0\_RESET\_SEL] Configures the reset type when the brownout detector is triggered.\\
0: Chip reset\\
1: System reset\\ 
(R/W)
\label{fielddesc:LPANABODMODE0RESETENA}\item [LP\_ANA\_BOD\_MODE0\_RESET\_ENA] Configures whether to enable reset for the brownout detector.\\
0: Disable \\
1: Enable\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}
