\begin{register}{H}{LP\_AON\_SYS\_CFG\_REG}{0x0034}\label{regdesc:LPAONSYSCFGREG}
\regfield{LP\_AON\_HPSYS\_SW\_RESET}{1}{31}{{0}}%
\regfield{LP\_AON\_FORCE\_DOWNLOAD\_BOOT}{1}{30}{{0}}%
\regfield{(reserved)}{30}{0}{000000000000000000000000000000}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:LPAONFORCEDOWNLOADBOOT}\item [LP\_AON\_FORCE\_DOWNLOAD\_BOOT] Configures whether to trigger a CPU reset and switch the chip boot mode.\\
0: No effect.\\
1: If \hyperref[fielddesc:EFUSEDISFORCEDOWNLOAD]{EFUSE\_DIS\_FORCE\_DOWNLOAD} is 0, software can force switch the chip from SPI Boot mode to Joint Download Boot mode and trigger a CPU reset.\\
(R/W)
\label{fielddesc:LPAONHPSYSSWRESET}\item [LP\_AON\_HPSYS\_SW\_RESET] Configures System Reset. \\
0: No effect\\
1: Enable reset\\
(WT)
\end{reglist}\end{regdesc}
\end{register}


\begin{register}{H}{LP\_AON\_CPUCORE0\_CFG\_REG}{0x0038}\label{regdesc:LPAONCPUCORE0CFGREG}

\regfield{LP\_AON\_CORE0\_STAT\_VECTOR\_SEL}{1}{30}{{1}}%
\regfield{(reserved)}{1}{29}{0}%
\regfield{LP\_AON\_CPU\_CORE0\_SW\_RESET}{1}{28}{{0}}%
\regfield{(reserved)}{28}{0}{0000000000000000000000000000}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:LPAONCPUCORE0SWRESET}\item [LP\_AON\_CPU\_CORE0\_SW\_RESET] Configures CPU Reset. \\
0: No effect\\
1: Enable reset\\
(WT)
\label{fielddesc:LPAONCORE0STATVECTORSEL}\item [LP\_AON\_CORE0\_STAT\_VECTOR\_SEL] Configures whether to start up the CPU from the RTC fast memory.\\
0: Do not start up from the RTC fast memory.\\
1: Start up from the RTC fast memory.\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}
