
\begin{register}{H}{pma\_cfgX}{0xBC0-0xBCF}
\label{regdesc:pma_cfgX}
\regfield{A}{2}{30}{{2}}%
\regfield{LOCK}{1}{29}{{0}}%
\regfield{reserved}{1}{28}{{0}}%
\regfield{ATTRIBUTE}{4}{24}{{0}}%
\regfield{reserved}{19}{5}{{0}}%
\regfield{READ}{1}{4}{{0}}%
\regfield{WRITE}{1}{3}{{0}}%
\regfield{EXECUTE}{1}{2}{{0}}%
\regfield{reserved}{1}{1}{{0}}%
\regfield{TYPE}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:A}\item [A] Configures address type. The functionality is the same as pmpcfg register's A field. (R/W)\\
0x0: OFF\\
0x1: TOR\\
0x2: NA4\\
0x3: NAPOT\\
\label{fielddesc:LOCK}\item [LOCK] Configures whether to lock the corresponding pma\_cfgX and pma\_addrX. (R/W)\\
0: Not locked\\
1: Locked. The write permission to the corresponding pma\_cfgX and pma\_addrX is revoked.\\
It can only be unlocked by core reset.
\label{fielddesc:ATTRIBUTE}\item [ATTRIBUTE] Configures the values to be driven on DRAM attribute ports. (R/W)
\label{fielddesc:READ}\item [READ] Configures read-permission for the corresponding region.\\
0: Read not allowed\\
1: Read allowed\\
 (R/W)
\label{fielddesc:WRITE}\item [WRITE] Configures write-permission for the corresponding region.\\
0: Write not allowed\\
1: Write allowed\\
 (R/W)
\label{fielddesc:EXECUTE}\item [EXECUTE] Configures execute-permission for the corresponding region.\\
0: Execution not allowed\\
1: Execution allowed\\
 (R/W)
\label{fielddesc:TYPE}\item [TYPE] Configures region type. (R/W)\\
0x0: Invalid memory region (RWX access will be treated as 0, even if programmed to 1)\\
0x1: Valid memory region (Programmed RWX access will be applicable)\\
\end{reglist}
\end{regdesc}
\end{register}
