\begin{register}{H}{DS\_QUERY\_KEY\_WRONG\_REG}{0x0E10}\label{regdesc:DSQUERYKEYWRONGREG}
\regfield{(reserved)}{28}{4}{0000000000000000000000000000}%
\regfield{DS\_QUERY\_KEY\_WRONG}{4}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\item [DS\_QUERY\_KEY\_WRONG] Represents the specific problem with HMAC initialization.\\
0: HMAC is not called\\
1-15: HMAC was activated, but the RSA\_DS peripheral did not successfully receive the \begin{math}RSA\_DS\_KEY\end{math} from the HMAC peripheral. (The biggest value is 15)\\
(RO)
\end{reglist}\end{regdesc}
\end{register}
