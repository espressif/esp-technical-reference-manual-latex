\begin{register}{H}{mcause}{0x342}\label{regdesc:LPMCAUSE}
\regfield{Interrupt Flag}{1}{31}{{0}}%
\regfield{(reserved)}{26}{5}{{0x0000000}}%
\regfield{Exception Code}{5}{0}{{0x00}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:LPMCAUSECODE}\item [Exception Code] This field is automatically updated with unique ID of the most recent exception or interrupt due to which CPU entered trap. Possible exception IDs are:\\
0x2: Illegal instruction\\
0x3: Hardware breakpoint/watchpoint or EBREAK\\
0x6: Misaligned atomic instructions\\
Note: Exception ID 0x0 (instruction access misaligned) is not present because CPU always masks the lowest bit of the address during instruction fetch.\\
(R/W)
\label{fielddesc:LPMCAUSEINT}\item [Interrupt Flag] This flag is automatically updated when CPU enters trap. If this is found to be set, it indicates that the latest trap occurred due to an interrupt. For exceptions it remains unset.  (R/W)
\end{reglist}\end{regdesc}
\end{register}

\begin{register}{H}{mtval}{0x343}\label{regdesc:LPMTVAL}
\regfield{MTVAL}{32}{0}{{0x00000000}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:LPMTVAL}\item [MTVAL] Configures machine trap value. This is automatically updated with an exception dependent data which may be useful for handling that exception. Data is to be interpreted depending upon exception IDs:\\
0x1: Faulting virtual address of instruction\\
0x2: Faulting instruction opcode\\
0x5: Faulting data address of load operation\\
0x7: Faulting data address of store operation\\
Note: The value of this register is not valid for other exception IDs and interrupts.\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}
