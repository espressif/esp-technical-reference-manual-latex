\begin{register}{H}{PAU\_REGDMA\_CONF\_REG}{0x0000}\label{regdesc:PAUREGDMACONFREG}
\regfield{(reserved)}{20}{12}{00000000000000000000}%
\regfield{PAU\_SEL\_MAC}{1}{11}{{0}}%
\regfield{PAU\_TO\_MEM\_MAC}{1}{10}{{0}}%
\regfield{PAU\_START\_MAC}{1}{9}{{0}}%
\regfield{PAU\_LINK\_SEL}{4}{5}{{0x0}}%
\regfield{PAU\_TO\_MEM}{1}{4}{{0}}%
\regfield{PAU\_START}{1}{3}{{0}}%
\regfield{PAU\_FLOW\_ERR}{3}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}\begin{reglist}
\label{fielddesc:PAUFLOWERR}\item [PAU\_FLOW\_ERR] Represents the backup error type.\\
0: Received software triggered interrupt clear signal\\
1: Peripheral AHB bus timeout\\
2: Memory AHB bus timeout\\
3: Address link list wait timer timeout\\
4: Address mapping error\\
(RO)
\label{fielddesc:PAUSTART}\item [PAU\_START] Write 1 to initiate the backup start signal. (WT)
\label{fielddesc:PAUTOMEM}\item [PAU\_TO\_MEM] Configures the backup direction.\\
0: register to memory\\
1: memory to register\\
(R/W)
\label{fielddesc:PAULINKSEL}\item [PAU\_LINK\_SEL] Configures to select the link.\\
0: PAU\_REGDMA\_LINK\_0\_ADDR\_REG\\
1: PAU\_REGDMA\_LINK\_1\_ADDR\_REG\\
2: PAU\_REGDMA\_LINK\_2\_ADDR\_REG\\
3: PAU\_REGDMA\_LINK\_3\_ADDR\_REG\\
(R/W)
\label{fielddesc:PAUSTARTMAC}\item [PAU\_START\_MAC] Write 1 to initiate the MAC software backup start signal. (WT)
\label{fielddesc:PAUTOMEMMAC}\item [PAU\_TO\_MEM\_MAC] Configures the MAC software backup direction.\\
0: register to memory\\
1: memory to register\\
(R/W)
\label{fielddesc:PAUSELMAC}\item [PAU\_SEL\_MAC] Configures the MAC hardware/software selection. \\
0: select the hardware path\\
1: select the software path\\
(R/W)
\end{reglist}\end{regdesc}
\end{register}
