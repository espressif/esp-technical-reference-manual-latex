\begin{register}{H}{{mcliccfg}}{0x20800000}\label{regdesc:MCLICCFG_REG}
\regfield{(reserved)}{26}{6}{{0x00000}}%
\regfield{NMBITS}{2}{4}{{0x0}}%
\regfield{MNLBITS}{4}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:MCLICCFG_MNLBITS}\item [MNLBITS] Configures the number of upper bits that are used globally (for HP core) to encode the interrupt level in all the 8-bit \hyperref[regdesc:MCLICINTCTL_REG_i]{clicintctl[i]} registers for each machine mode interrupt.\\
The only allowed values are in the range 0-8. Any higher value will be clamped to 8. \\
(R/W)
\label{fielddesc:MCLICCFG_NMBITS}\item [NMBITS] Configures the number of bits in \hyperref[fielddesc:MCLICINTATTR_i_MODE]{clicintattr[i].MODE} to be used globally (for HP core) to represent the mode of an interrupt. For systems with machine mode only, this is hardwired to 0. This indicates that the \hyperref[fielddesc:MCLICINTATTR_i_MODE]{clicintattr[i].MODE} field is hardwired to 0x3 and therefore all interrupts are in machine mode only. (RO) \\
\end{reglist}
\end{regdesc}
\end{register}


\begin{register}{H}{{clicinfo}}{0x20800004}\label{regdesc:CLICINFO_REG}
\regfield{(reserved)}{7}{25}{{0x00}}%
\regfield{CLICINTCTLBITS}{4}{21}{{0x3}}%
\regfield{(reserved)}{8}{13}{{0x00}}%
\regfield{NUM\_INTERRUPTS}{13}{0}{{0x0030}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:CLICINFO_NUM_INTERRUPTS}\item [NUM\_INTERRUPTS] Represents the number of interrupts supported by this implementation of CLIC. Hardwired to 48. (RO)
\label{fielddesc:CLICINFO_CLICINTCTLBITS}\item [CLICINTCTLBITS] Represents the number of higher bits which are actually programmable in the 8-bit \hyperref[regdesc:MCLICINTCTL_REG_i]{clicintctl[i]} registers. Hardwired to 0x3. (RO)
\end{reglist}
\end{regdesc}
\end{register}


\begin{register}{H}{{clicintip[i]}}{0x20801000 + 4*i}\label{regdesc:MCLICINTIP_REG_i}
\regfield{(reserved)}{7}{1}{{0x00}}%
\regfield{{IP[i]}}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:MCLICINTIP_i}\item [{IP[i]}] Represents the pending state for i\textsuperscript{th} CLIC machine mode interrupt. \\
0: i\textsuperscript{th} interrupt is not pending \\
1: i\textsuperscript{th} interrupt is pending \\
\begin{itemize}
    \item When the i\textsuperscript{th} interrupt is configured as level type using \hyperref[fielddesc:MCLICINTATTR_i_TRIG]{clicintattr[i].TRIG}, this bit is read-only, and to clear it the interrupt must be cleared from source.
    \item When the i\textsuperscript{th} interrupt is configured as edge triggered using \hyperref[fielddesc:MCLICINTATTR_i_TRIG]{clicintattr[i].TRIG}, this bit is R/W, and thus it is to be cleared by writing 0 to this register.
\end{itemize}
(R/W)
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{{clicintie[i]}}{0x20801001 + 4*i}\label{regdesc:MCLICINTIE_REG_i}
\regfield{(reserved)}{7}{1}{{0x00}}%
\regfield{{IE[i]}}{1}{0}{{0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:MCLICINTIE_i}\item [{IE[i]}] Configures whether to enable the i\textsuperscript{th} CLIC machine mode interrupt.\\
0: Disable \\
1: Enabled \\
(R/W) \\
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{{clicintattr[i]}}{0x20801002 + 4*i}\label{regdesc:MCLICINTATTR_REG_i}
\regfield{MODE}{2}{6}{{0x3}}%
\regfield{(reserved)}{3}{3}{{0x0}}%
\regfield{TRIG}{2}{1}{{0x0}}%
\regfield{SHV}{1}{0}{{0x0}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:MCLICINTATTR_i_SHV}\item [{SHV}] Configures hardware vectoring for the i\textsuperscript{th} CLIC machine mode interrupt. \\
0: i\textsuperscript{th} interrupt is not hardware vectored. It means that upon taking this interrupt, the CPU will jump to the address configured in \hyperref[fielddesc:MTVECBASE]{mtvec}. \\
1: i\textsuperscript{th} interrupt is hardware vectored. It means that upon taking this interrupt, the CPU will jump to the word address stored at (mtvt + 4*i) relative to the base address configured in \hyperref[fielddesc:MTVTBASE]{mtvt}. \\
(R/W)
\label{fielddesc:MCLICINTATTR_i_TRIG}\item [{TRIG}] Configures the trigger type and polarity of the i\textsuperscript{th} CLIC machine mode interrupt. \\
0x0: interrupt is positive level-triggered \\
0x1: interrupt is positive edge-triggered \\
0x2: interrupt is negative level-triggered \\
0x3: interrupt is negative edge-triggered \\
(R/W)
\label{fielddesc:MCLICINTATTR_i_MODE}\item [{MODE}] Configures the privilege mode of the i\textsuperscript{th} CLIC interrupt. \\
0x3: machine mode \\
0x0: user mode \\
(R/W)
\end{reglist}
\end{regdesc}
\end{register}

\begin{register}{H}{{clicintctl[i]}}{0x20801003 + 4*i}\label{regdesc:MCLICINTCTL_REG_i}
\regfield{{CTL[i]}}{3}{5}{{0x0}}%
\regfield{(reserved)}{5}{0}{{0x1f}}%
\reglabel{Reset}\regnewline%
\begin{regdesc}
\begin{reglist}
\label{fielddesc:MCLICINTCTL_i}\item [{CTL[i]}] Configures the level and priority of the i\textsuperscript{th} CLIC machine mode interrupt. \\
\begin{itemize}
    \item Since the \hyperref[fielddesc:CLICINFO_CLICINTCTLBITS]{CLICINTCTLBITS} parameter is 3 in this implementation of CLIC, therefore the actual number of bits implemented in \hyperref[regdesc:MCLICINTCTL_REG_i]{clicintctl[i]} is the same, and thus the lower 5 bits of this register are hardwired to 1. \\
    \item The configured value of \hyperref[fielddesc:MCLICCFG_MNLBITS]{mcliccfg.MNLBITS} determines the number of higher bits in \hyperref[regdesc:MCLICINTCTL_REG_i]{clicintctl[i]} which encode the level of the interrupt, while the remaining lower bits encode the priority.
\end{itemize}
(R/W)
\end{reglist}
\end{regdesc}
\end{register}
